1import { createContext, useContext, useState, useEffect } from 'react';
2import api from '../api/axiosConfig';
3import { useUser } from './UserContext';
4
5const AddressContext = createContext();
6
7export function AddressProvider({ children }) {
8  const [addresses, setAddresses] = useState([]);
9  const [loading, setLoading] = useState(false);
10  const { user } = useUser();
11
12  useEffect(() => {
13    if (user) {
14      fetchAddresses();
15    } else {
16      setAddresses([]);
17    }
18  }, [user]);
19
20  const fetchAddresses = async () => {
21    if (!user) return;
22    
23    try {
24      setLoading(true);
25      const response = await api.get('/addresses/');
26      setAddresses(Array.isArray(response.data) ? response.data : []);
27    } catch (error) {
28      console.error('Failed to fetch addresses', error);
29      setAddresses([]);
30    } finally {
31      setLoading(false);
32    }
33  };
34
35  const createAddress = async (addressData) => {
36    if (!user) {
37      throw new Error('Please login to add addresses');
38    }
39
40    try {
41      const response = await api.post('/addresses/', addressData);
42      await fetchAddresses();
43      return response.data;
44    } catch (error) {
45      console.error('Failed to create address', error);
46      throw error;
47    }
48  };
49
50  const updateAddress = async (addressId, addressData) => {
51    if (!user) return;
52
53    try {
54      const response = await api.put(`/addresses/${addressId}/`, addressData);
55      await fetchAddresses();
56      return response.data;
57    } catch (error) {
58      console.error('Failed to update address', error);
59      throw error;
60    }
61  };
62
63  const deleteAddress = async (addressId) => {
64    if (!user) return;
65
66    try {
67      await api.delete(`/addresses/${addressId}/`);
68      await fetchAddresses();
69    } catch (error) {
70      console.error('Failed to delete address', error);
71      throw error;
72    }
73  };
74
75  const getAddressById = async (addressId) => {
76    if (!user) return null;
77
78    try {
79      const response = await api.get(`/addresses/${addressId}/`);
80      return response.data;
81    } catch (error) {
82      console.error('Failed to fetch address', error);
83      throw error;
84    }
85  };
86
87  const value = {
88    addresses,
89    loading,
90    fetchAddresses,
91    createAddress,
92    updateAddress,
93    deleteAddress,
94    getAddressById,
95  };
96
97  return (
98    <AddressContext.Provider value={value}>
99      {children}
100    </AddressContext.Provider>
101  );
102}
103
104export const useAddresses = () => useContext(AddressContext);
