1import { createSlice, createAsyncThunk } from '@reduxjs/toolkit'
2import api from '../../api/axiosConfig'
3
4// Async thunks
5export const loginUser = createAsyncThunk(
6  'user/login',
7  async ({ email, password }, { rejectWithValue }) => {
8    try {
9      const response = await api.post('/login/', { email, password })
10      const { access, refresh } = response.data
11      localStorage.setItem('access_token', access)
12      localStorage.setItem('refresh_token', refresh)
13      
14      const userResponse = await api.get('/me/')
15      localStorage.setItem('user', JSON.stringify(userResponse.data))
16      
17      // Try to fetch profile
18      let profile = null
19      try {
20        const profileResponse = await api.get('/profile/')
21        profile = profileResponse.data
22      } catch (profileError) {
23        console.log('Profile not found')
24      }
25      
26      return { user: userResponse.data, profile }
27    } catch (error) {
28      return rejectWithValue(error.response?.data || 'Login failed')
29    }
30  }
31)
32
33export const fetchUserProfile = createAsyncThunk(
34  'user/fetchProfile',
35  async (_, { rejectWithValue }) => {
36    try {
37      const response = await api.get('/profile/')
38      return response.data
39    } catch (error) {
40      return rejectWithValue(error.response?.data || 'Failed to fetch profile')
41    }
42  }
43)
44
45const userSlice = createSlice({
46  name: 'user',
47  initialState: {
48    user: JSON.parse(localStorage.getItem('user')) || null,
49    profile: null,
50    loading: false,
51    error: null
52  },
53  reducers: {
54    setUser: (state, action) => {
55      state.user = action.payload
56    },
57    setProfile: (state, action) => {
58      state.profile = action.payload
59    },
60    logoutUser: (state) => {
61      state.user = null
62      state.profile = null
63      localStorage.removeItem('access_token')
64      localStorage.removeItem('refresh_token')
65      localStorage.removeItem('user')
66    },
67    clearError: (state) => {
68      state.error = null
69    }
70  },
71  extraReducers: (builder) => {
72    builder
73      .addCase(loginUser.pending, (state) => {
74        state.loading = true
75        state.error = null
76      })
77      .addCase(loginUser.fulfilled, (state, action) => {
78        state.loading = false
79        state.user = action.payload.user
80        state.profile = action.payload.profile
81      })
82      .addCase(loginUser.rejected, (state, action) => {
83        state.loading = false
84        state.error = action.payload
85      })
86      .addCase(fetchUserProfile.fulfilled, (state, action) => {
87        state.profile = action.payload
88      })
89  }
90})
91
92export const { setUser, setProfile, logoutUser, clearError } = userSlice.actions
93
94// Selectors
95export const selectUser = (state) => state.user.user
96export const selectProfile = (state) => state.user.profile
97export const selectUserLoading = (state) => state.user.loading
98export const selectUserError = (state) => state.user.error
99
100export default userSlice.reducer
