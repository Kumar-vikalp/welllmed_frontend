1import { Routes, Route, useLocation } from 'react-router-dom';
2import { Navigate } from 'react-router-dom';
3import { AnimatePresence, motion } from 'framer-motion';
4import { useSelector } from 'react-redux';
5import { selectUser } from './store/slices/userSlice';
6import { Suspense, useEffect } from 'react';
7import { useDispatch } from 'react-redux';
8import { fetchCart } from './store/slices/cartSlice';
9
10import Home from './pages/Home';
11import Login from './pages/Login';
12import Signup from './pages/Signup';
13import ProductsPage from './pages/ProductsPage';
14import GenericInfoPage from './pages/GenericInfoPage';
15import ForgotPasswordPage from './pages/ForgotPasswordPage';
16import ResetPasswordPage from './pages/ResetPasswordPage';
17import ProductDetails from './pages/ProductDetails';
18import CartPage from './pages/CartPage';
19import CheckoutPage from './pages/CheckoutPage';
20import SearchPage from './pages/SearchPage';
21import ProfilePage from './pages/ProfilePage';
22import OrderHistoryPage from './pages/OrderHistoryPage';
23import OrderSuccessPage from './pages/OrderSuccessPage';
24import AboutPage from './pages/AboutPage';
25import ContactPage from './pages/ContactPage';
26import HelpPage from './pages/HelpPage';
27import HowToOrderPage from './pages/HowToOrderPage';
28import LabTestsPage from './pages/LabTestsPage';
29import PrescriptionUploadPage from './pages/PrescriptionUploadPage';
30
31import Layout from './components/Layout';
32import ProtectedRoute from './components/ProtectedRoute';
33
34// Loading component
35const PageLoader = () => (
36  <motion.div 
37    initial={{ opacity: 0 }}
38    animate={{ opacity: 1 }}
39    className="flex items-center justify-center min-h-screen"
40  >
41    <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-purple-600"></div>
42  </motion.div>
43);
44
45export default function App() {
46  const location = useLocation();
47  const dispatch = useDispatch();
48  const user = useSelector(selectUser);
49
50  // Initialize cart when user is available
51  useEffect(() => {
52    if (user) {
53      dispatch(fetchCart());
54    }
55  }, [user, dispatch]);
56
57  return (
58    <Suspense fallback={<PageLoader />}>
59      <Routes location={location} key={location.pathname}>
60        {/* Public Routes with Layout */}
61        <Route path="/" element={<Layout />}>
62          <Route index element={<Home />} />
63          <Route path="products" element={<ProductsPage />} />
64          <Route path="search" element={<SearchPage />} />
65          <Route path="generic-info" element={<GenericInfoPage />} />
66          <Route path="about" element={<AboutPage />} />
67          <Route path="contact" element={<ContactPage />} />
68          <Route path="help" element={<HelpPage />} />
69          <Route path="how-to-order" element={<HowToOrderPage />} />
70          <Route path="lab" element={<LabTestsPage />} />
71          <Route path="upload-prescription" element={<PrescriptionUploadPage />} />
72          <Route path="product/:slug" element={<ProductDetails />} />
73          <Route path="cart" element={<CartPage />} />
74          
75          {/* Protected Routes with Layout */}
76          <Route path="checkout" element={<ProtectedRoute><CheckoutPage /></ProtectedRoute>} />
77          <Route path="profile" element={<ProtectedRoute><ProfilePage /></ProtectedRoute>} />
78          <Route path="orders" element={<ProtectedRoute><OrderHistoryPage /></ProtectedRoute>} />
79        </Route>
80        
81        {/* Routes without the main layout */}
82        <Route path="/login" element={user ? <Navigate to="/" replace /> : <Login />} />
83        <Route path="/signup" element={user ? <Navigate to="/" replace /> : <Signup />} />
84        <Route path="/forgot-password" element={user ? <Navigate to="/" replace /> : <ForgotPasswordPage />} />
85        <Route path="/reset-password" element={user ? <Navigate to="/" replace /> : <ResetPasswordPage />} />
86        <Route path="/order-success" element={<ProtectedRoute><OrderSuccessPage /></ProtectedRoute>} />
87      </Routes>
88    </Suspense>
89  );
90}
