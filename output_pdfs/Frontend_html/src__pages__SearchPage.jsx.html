1import { useEffect, useState } from 'react'
2import { useSearchParams } from 'react-router-dom'
3import api from '../api/axiosConfig'
4import ProductCard from '../components/ProductCard'
5import Skeleton from '../components/Skeleton'

8export default function SearchPage() {
9  const [searchParams] = useSearchParams()
10  const [products, setProducts] = useState([])
11  const [loading, setLoading] = useState(true)
12  const [searchQuery, setSearchQuery] = useState(searchParams.get('q') || '')
13  const [recentSearches] = useState(['PARACETAMOL', 'VITAMIN C', 'COUGH SYRUP', 'ANTIBIOTICS'])

15    'FEVER & PAIN', 'ALLERGY & COLD', 'DIGESTIVE HEALTH', 'VITAMINS & SUPPLEMENTS'
16  ])
17
18  useEffect(() => {
19    const query = searchParams.get('q')
20    if (query) {
21      setSearchQuery(query)
22      searchProducts(query)
23    } else {
24      setLoading(false)
25    }
26  }, [searchParams])
27
28  const searchProducts = async (query) => {
29    setLoading(true)
30    try {
31      const response = await api.get(`/products/?search=${encodeURIComponent(query)}`)
32      const data = response.data
33      const productsData = data.results || (Array.isArray(data) ? data : [data])
34      
35      const transformedProducts = productsData.map(transformProduct)
36      setProducts(transformedProducts)
37    } catch (error) {
38      console.error('Failed to search products', error)
39      setProducts([])
40    } finally {
41      setLoading(false)
42    }
43  }
44
45  const transformProduct = (product) => ({
46    product_id: product.product_id,
47    slug: product.slug,
48    name: product.name,
49    company: product.company,
50    disease_category: product.disease_category,
51    mrp: parseFloat(product.mrp),
52    discount: product.discount,
53    discounted_price: parseFloat(product.discounted_price),
54    images: product.images?.map(img => img.stream_url) || ['/images/placeholder.jpg'],
55    trending: product.trending,
56    available_stock: product.available_stock || 100,
57    description: product.description || `${product.name} from ${product.company}. Effective treatment for ${product.disease_category}.`
58  })
59
60  const handleSearch = (query) => {
61    if (query.trim()) {
62      const newUrl = `/search?q=${encodeURIComponent(query.trim())}`
63      window.history.pushState({}, '', newUrl)
64      // Update the search params to trigger useEffect
65      const newSearchParams = new URLSearchParams(newUrl.split('?')[1])
66      setSearchParams(newSearchParams)
67      searchProducts(query.trim())
68    }
69  }
70
71  return (
72    <motion.div

74      animate={{ opacity: 1 }} 
75      className="min-h-screen bg-[#FFFDF5]"
76    >
77      {/* Background Pattern */}
78      <div 
79        className="absolute inset-0 opacity-5"
80        style={{
81          backgroundImage: 'radial-gradient(#000 1.5px, transparent 1.5px)',
82          backgroundSize: '20px 20px'
83        }}
84      ></div>
85
86      {/* Search Header */}

88        <div className="sticky top-16 z-50 bg-[#C4B5FD] border-b-8 border-black">
89        <div className="max-w-7xl mx-auto p-4 sm:p-6">
90          <div className="relative">
91            <input
92              type="text"
93              value={searchQuery}
94              onChange={(e) => setSearchQuery(e.target.value)}
95              onKeyPress={(e) => e.key === 'Enter' && handleSearch(searchQuery)}

98                placeholder:text-black/40 placeholder:font-bold placeholder:uppercase
99                focus-visible:bg-[#FFD93D] focus-visible:shadow-[4px_4px_0px_0px_#000]
100                focus-visible:outline-none focus-visible:ring-0
101                transition-all duration-100"
102            />
103            <button 
104              onClick={() => handleSearch(searchQuery)}
105              className="absolute left-4 top-1/2 transform -translate-y-1/2 hover:scale-110 transition-transform duration-100"
106            >
107              <svg className="w-6 h-6 stroke-[3px]" fill="none" stroke="currentColor" viewBox="0 0 24 24">
108                <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="3" d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z" />
109              </svg>
110            </button>
111          </div>
112        </div>
113      </div>
114
115      <div className="relative z-10 max-w-7xl mx-auto p-4 sm:p-8">
116        {!searchQuery ? (
117          /* Search Suggestions */
118          <div className="space-y-8">
119            {/* Header */}
120            <div className="text-center">
121              <h1 className="text-3xl sm:text-5xl md:text-7xl font-black uppercase tracking-tight mb-4 -rotate-1">
122                SEARCH MEDICINES
123              </h1>
124              <div className="bg-[#FFD93D] border-4 border-black shadow-[6px_6px_0px_0px_#000] p-4 rotate-1 inline-block">
125                <p className="font-bold uppercase tracking-wide">FIND EXACTLY WHAT YOU NEED</p>
126              </div>
127            </div>
128
129            {/* Recent Searches */}
130            <div className="bg-white border-4 border-black shadow-[8px_8px_0px_0px_#000] overflow-hidden">
131              <div className="bg-[#FF6B6B] border-b-4 border-black p-4 sm:p-6">
132                <h2 className="font-black uppercase tracking-widest text-sm text-white">RECENT SEARCHES</h2>
133              </div>
134              <div className="p-4 sm:p-6">
135                <div className="flex flex-wrap gap-3">
136                  {recentSearches.map((search, index) => (
137                    <button
138                      key={index}
139                      onClick={() => handleSearch(search)}
140                      className="bg-white border-4 border-black font-bold uppercase tracking-wide text-sm px-4 py-3 h-12

143                        active:shadow-none active:translate-x-[4px] active:translate-y-[4px]
144                        transition-all duration-100"
145                    >
146                      {search}
147                    </button>
148                  ))}
149                </div>
150              </div>
151            </div>
152
153            {/* Popular Categories */}
154            <div className="bg-white border-4 border-black shadow-[8px_8px_0px_0px_#000] overflow-hidden">

161                    <button
162                      key={index}
163                      onClick={() => handleSearch(category)}
164                      className="bg-white border-4 border-black p-4 sm:p-6 text-left shadow-[4px_4px_0px_0px_#000]

178            </div>
179
180            {/* Explore More */}

189                onClick={() => window.location.href = '/products'}
190                className="bg-[#FF6B6B] border-4 border-white font-black uppercase tracking-widest

192                  shadow-[6px_6px_0px_0px_#fff]
193                  hover:shadow-[3px_3px_0px_0px_#fff] hover:translate-x-[3px] hover:translate-y-[3px]
194                  active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
195                  transition-all duration-100"
196              >
197                BROWSE ALL PRODUCTS
198              </button>
199            </div>
200          </div>
201        ) : (
202          /* Search Results */
203          <div>
204            {/* Results Header */}
205            <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center gap-4 mb-8">
206              <div>
207                <h2 className="text-2xl sm:text-3xl md:text-4xl font-black uppercase tracking-tight mb-2">
208                  SEARCH RESULTS
209                </h2>
210                <div className="bg-[#C4B5FD] border-4 border-black shadow-[4px_4px_0px_0px_#000] p-3 -rotate-1 inline-block">
211                  <p className="font-bold uppercase tracking-wide text-sm">
212                    FOR "{searchQuery}" - {!loading && `${products.length} PRODUCTS FOUND`}
213                  </p>
214                </div>
215              </div>
216              
217              <button
218                onClick={() => {
219                  setSearchQuery('')
220                  window.history.pushState({}, '', '/search')
221                }}
222                className="bg-white border-4 border-black font-black uppercase tracking-widest 
223                  text-xs px-4 py-3 h-12 w-full sm:w-auto
224                  shadow-[4px_4px_0px_0px_#000]
225                  hover:bg-[#FFD93D] hover:shadow-[2px_2px_0px_0px_#000] hover:translate-x-[2px] hover:translate-y-[2px]
226                  active:shadow-none active:translate-x-[4px] active:translate-y-[4px]
227                  transition-all duration-100"
228              >
229                CLEAR SEARCH
230              </button>
231            </div>
232
233            {loading ? (
234              <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5 gap-4">
235                {Array(10).fill(0).map((_, i) => (
236                  <div key={i} className="bg-white border-4 border-black shadow-[6px_6px_0px_0px_#000] h-80"></div>
237                ))}
238              </div>
239            ) : products.length > 0 ? (
240              <div className="grid grid-cols-2 sm:grid-cols-3 lg:grid-cols-4 xl:grid-cols-5 gap-4">
241                {products.map(product => (
242                  <ProductCard key={product.product_id} product={product} />
243                ))}
244              </div>
245            ) : (
246              /* No Results */
247              <div className="text-center py-16">
248                <div className="bg-white border-4 border-black shadow-[12px_12px_0px_0px_#000] p-8 sm:p-12 inline-block rotate-1">
249                  <div className="w-24 h-24 bg-[#FF6B6B] border-4 border-black mx-auto mb-6 flex items-center justify-center -rotate-12">
250                    <svg className="w-12 h-12 stroke-[4px] text-white" fill="none" stroke="currentColor" viewBox="0 0 24 24">
251                      <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="4" d="M21 21l-6-6m2-5a7 7 0 11-14 0 7 7 0 0114 0z" />

255                  <p className="font-bold uppercase tracking-wide mb-6">TRY SEARCHING WITH DIFFERENT KEYWORDS</p>
256                  <button
257                    onClick={() => {
258                      setSearchQuery('')
259                      window.history.pushState({}, '', '/search')
260                    }}
261                    className="bg-[#FFD93D] border-4 border-black font-black uppercase tracking-widest 
262                      text-sm px-6 py-4 h-14
263                      shadow-[6px_6px_0px_0px_#000]
264                      hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
265                      active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
266                      transition-all duration-100"
267                  >
268                    CLEAR SEARCH
269                  </button>
270                </div>
271              </div>
272            )}
273
274            {/* Search Tips */}
275            {!loading && products.length === 0 && (
276              <div className="mt-12">
277                <div className="bg-[#C4B5FD] border-4 border-black shadow-[8px_8px_0px_0px_#000] overflow-hidden">
278                  <div className="bg-black border-b-4 border-black p-4 sm:p-6">
279                    <h3 className="font-black uppercase tracking-widest text-sm text-white">SEARCH TIPS</h3>
280                  </div>
281                  <div className="p-4 sm:p-6">
282                    <div className="grid grid-cols-1 sm:grid-cols-2 gap-4">
283                      <div className="space-y-3">
284                        <div className="flex items-center gap-3">
285                          <div className="w-2 h-2 bg-black"></div>
286                          <span className="font-bold text-sm uppercase">TRY GENERIC NAMES</span>
287                        </div>

289                          <div className="w-2 h-2 bg-black"></div>
290                          <span className="font-bold text-sm uppercase">CHECK SPELLING</span>
291                        </div>
292                        <div className="flex items-center gap-3">
293                          <div className="w-2 h-2 bg-black"></div>
294                          <span className="font-bold text-sm uppercase">USE FEWER WORDS</span>
295                        </div>
