8// Async thunks
9export const fetchCart = createAsyncThunk(
10  'cart/fetchCart',
11  async (_, { rejectWithValue }) => {
12    try {
13      const response = await api.get('/cart/')
14      const cartItems = Array.isArray(response.data) ? response.data : [response.data]
15      return cartItems.map(item => ({
16        product_id: item.product.product_id,
17        cart_item_id: item.cart_item_id,
18        name: item.product.name,
19        company: item.product.company,
20        disease_category: item.product.disease_category,
21        mrp: parseFloat(item.product.mrp),
22        discount: item.product.discount,
23        images: item.product.images?.map(img => img.stream_url) || ['/images/placeholder.jpg'],
24        trending: item.product.trending,
25        slug: item.product.slug,
26        qty: item.quantity,
27        available_stock: item.product.available_stock || 100,
28        discounted_price: parseFloat(item.product.discounted_price)
29      }))
30    } catch (error) {
31      // If cart is empty or not found, return empty array instead of error
32      if (error.response?.status === 404 || error.response?.status === 400) {
33        return []
34      }
35      return rejectWithValue(error.response?.data || 'Failed to fetch cart')
36    }
37  }
38)

40export const addToCart = createAsyncThunk(
41  'cart/addToCart',
42  async ({ product_id, quantity = 1 }, { rejectWithValue }) => {
43    try {
44      const response = await api.post('/cart/add/', {
45        product_id,
46        quantity
47      })
48      return response.data
49    } catch (error) {
50      return rejectWithValue(error.response?.data || 'Failed to add to cart')
51    }
52  }
53)
54
55export const updateCartItem = createAsyncThunk(
56  'cart/updateCartItem',
57  async ({ cart_item_id, quantity }, { rejectWithValue }) => {
58    try {
59      const response = await api.patch(`/cart/update/${cart_item_id}/`, {
60        quantity
61      })
62      return response.data
63    } catch (error) {
64      return rejectWithValue(error.response?.data || 'Failed to update cart item')
65    }
66  }
67)
68
69export const removeCartItem = createAsyncThunk(
70  'cart/removeItem',
71  async (cart_item_id, { rejectWithValue }) => {
72    try {
73      await api.delete(`/cart/remove/${cart_item_id}/`)
74      return cart_item_id
75    } catch (error) {
76      return rejectWithValue(error.response?.data || 'Failed to remove item')
77    }
78  }
79)

81const cartSlice = createSlice({
82  name: 'cart',
83  initialState: {
84    items: [],
85    loading: false,
86    syncing: false,
87    error: null,
88    lastSyncTime: null,
89    initialized: false
90  },
91  reducers: {
92    clearCart: (state) => {
93      state.items = []
94      state.initialized = false
95    },
96    
97    setInitialized: (state, action) => {
98      state.initialized = action.payload
99    }
100  },
101  extraReducers: (builder) => {
102    builder
103      .addCase(fetchCart.pending, (state) => {
104        state.loading = true
105        state.error = null
106      })
107      .addCase(fetchCart.fulfilled, (state, action) => {
108        state.loading = false
109        state.items = action.payload
110        state.initialized = true
111      })
112      .addCase(fetchCart.rejected, (state, action) => {
113        state.loading = false
114        state.error = action.payload
115        state.initialized = true
116      })
117      .addCase(addToCart.pending, (state) => {
118        state.syncing = true
119        state.error = null
120      })
121      .addCase(addToCart.fulfilled, (state, action) => {
122        state.syncing = false
123        // Refresh cart after adding
124        // The actual cart update will be handled by fetchCart
125      })
126      .addCase(addToCart.rejected, (state, action) => {
127        state.syncing = false
128        state.error = action.payload
129      })
130      .addCase(updateCartItem.pending, (state) => {
131        state.syncing = true
132      })
133      .addCase(updateCartItem.fulfilled, (state, action) => {
134        state.syncing = false
135        // Refresh cart after updating
136      })
137      .addCase(updateCartItem.rejected, (state, action) => {
138        state.syncing = false
139        state.error = action.payload
140      })
141      .addCase(removeCartItem.pending, (state) => {
142        state.syncing = true
143      })
144      .addCase(removeCartItem.fulfilled, (state, action) => {
145        state.syncing = false
146        state.items = state.items.filter(item => item.cart_item_id !== action.payload)
147      })
148      .addCase(removeCartItem.rejected, (state, action) => {
149        state.syncing = false
150        state.error = action.payload
151      })
152  }
153})
