1import { createContext, useContext, useState, useEffect } from 'react';
2import { useDispatch } from 'react-redux';
3import { setUser as setReduxUser, logoutUser as logoutReduxUser } from '../store/slices/userSlice';
4import api from '../api/axiosConfig';
5
6const UserContext = createContext();
7
8export function UserProvider({ children }) {
9  const dispatch = useDispatch();
10  const [user, setUser] = useState(null);
11  const [profile, setProfile] = useState(null);
12  const [loading, setLoading] = useState(true);
13
14  useEffect(() => {
15    const fetchUser = async () => {
16      const accessToken = localStorage.getItem('access_token');
17      if (accessToken) {
18        try {
19          const response = await api.get('/me/');
20          const userData = response.data;
21          setUser(userData);
22          dispatch(setReduxUser(userData));
23          // Fetch profile data
24          try {
25            const profileResponse = await api.get('/profile/');
26            setProfile(profileResponse.data);
27          } catch (profileError) {
28            console.log('Profile not found or incomplete');
29          }
30        } catch (error) {
31          console.error('Failed to fetch user on initial load', error);
32          localStorage.removeItem('access_token');
33          localStorage.removeItem('refresh_token');
34          localStorage.removeItem('user');
35          setUser(null);
36          setProfile(null);
37          dispatch(setReduxUser(null));
38        }
39      }
40      setLoading(false);
41    };
42    fetchUser();
43  }, [dispatch]);
44
45  const fetchProfile = async () => {
46    try {
47      const response = await api.get('/profile/');
48      setProfile(response.data);
49      return response.data;
50    } catch (error) {
51      console.error('Failed to fetch profile', error);
52      throw error;
53    }
54  };
55
56  const updateProfile = async (profileData) => {
57    try {
58      const response = await api.put('/profile/update/', profileData);
59      setProfile(response.data.profile);
60      return response.data;
61    } catch (error) {
62      console.error('Failed to update profile', error);
63      throw error;
64    }
65  };
66
67  const checkProfileStatus = async () => {
68    try {
69      const response = await api.get('/profile/status/');
70      return response.data;
71    } catch (error) {
72      console.error('Failed to check profile status', error);
73      throw error;
74    }
75  };
76
77  const getActiveTime = async () => {
78    try {
79      const response = await api.get('/profile/active-time/');
80      return response.data;
81    } catch (error) {
82      console.error('Failed to get active time', error);
83      throw error;
84    }
85  };
86
87  const signup = async (email, password) => {
88    await api.post('/signup/', { 
89      name: email.split('@')[0], // Use email prefix as default name
90      email, 
91      password,
92      confirmPassword: password 
93    });
94    await login(email, password);
95  };
96
97  const login = async (email, password) => {
98    const response = await api.post('/login/', { email, password });
99    const { access, refresh } = response.data;
100    localStorage.setItem('access_token', access);
101    localStorage.setItem('refresh_token', refresh);
102    const userResponse = await api.get('/me/');
103    setUser(userResponse.data);
104    dispatch(setReduxUser(userResponse.data));
105    // Try to fetch profile
106    try {
107      const profileResponse = await api.get('/profile/');
108      setProfile(profileResponse.data);
109    } catch (profileError) {
110      console.log('Profile not found');
111    }
112    localStorage.setItem('user', JSON.stringify(userResponse.data));
113  };
114
115  const logout = async () => {
116    const refreshToken = localStorage.getItem('refresh_token');
117    if (refreshToken) {
118        try {
119            await api.post('/logout/', { refresh: refreshToken });
120        } catch (error) {
121            console.error("Logout failed, proceeding to clear local data.", error);
122        }
123    }
124    localStorage.removeItem('access_token');
125    localStorage.removeItem('refresh_token');
126    localStorage.removeItem('user');
127    setUser(null);
128    setProfile(null);
129    dispatch(logoutReduxUser());
130  };
131  
132  const forgotPassword = async (email) => {
133    const response = await api.post('/forgot-password/', { email });
134    return response.data;
135  };
136
137  const resetPassword = async (email, otp, new_password) => {
138    const response = await api.post('/reset-password/', { email, otp, new_password });
139    return response.data;
140  };
141
142  const value = { 
143    user, 
144    profile, 
145    loading, 
146    signup, 
147    login, 
148    logout, 
149    fetchProfile,
150    updateProfile,
151    checkProfileStatus,
152    getActiveTime,
153    forgotPassword, 
154    resetPassword 
155  };
156
157  return (
158    <UserContext.Provider value={value}>
159      {!loading && children}
160    </UserContext.Provider>
161  );
162}
163
164export const useUser = () => useContext(UserContext);
