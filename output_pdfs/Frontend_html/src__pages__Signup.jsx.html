1import { useState } from 'react';
2import { Link, useNavigate } from 'react-router-dom';
3import { motion } from 'framer-motion';
4import { useUser } from '../contexts/UserContext';
5import { Pill, Eye, EyeOff, Star, Zap, Shield } from 'lucide-react';
6import api from '../api/axiosConfig';
7
8export default function Signup() {
9  const [formData, setFormData] = useState({
10    name: '',
11    email: '',
12    password: '',
13    confirmPassword: ''
14  });
15  const [showPassword, setShowPassword] = useState(false);
16  const [showConfirmPassword, setShowConfirmPassword] = useState(false);
17  const [errors, setErrors] = useState({});
18  const [isLoading, setIsLoading] = useState(false);
19  const { login } = useUser();
20  const navigate = useNavigate();
21
22  const handleChange = (e) => {
23    const { name, value } = e.target;
24    setFormData(prev => ({
25      ...prev,
26      [name]: value
27    }));
28    // Clear error when user starts typing
29    if (errors[name]) {
30      setErrors(prev => ({
31        ...prev,
32        [name]: ''
33      }));
34    }
35  };
36
37  const validateForm = () => {
38    const newErrors = {};
39
40    if (!formData.name.trim()) {
41      newErrors.name = 'Name is required';
42    }
43
44    if (!formData.email.trim()) {
45      newErrors.email = 'Email is required';
46    } else if (!/\S+@\S+\.\S+/.test(formData.email)) {
47      newErrors.email = 'Email is invalid';
48    }
49
50    if (!formData.password) {
51      newErrors.password = 'Password is required';
52    } else if (formData.password.length < 6) {
53      newErrors.password = 'Password must be at least 6 characters';
54    }
55
56    if (!formData.confirmPassword) {
57      newErrors.confirmPassword = 'Please confirm your password';
58    } else if (formData.password !== formData.confirmPassword) {
59      newErrors.confirmPassword = 'Passwords do not match';
60    }
61
62    setErrors(newErrors);
63    return Object.keys(newErrors).length === 0;
64  };
65
66  const handleSubmit = async (e) => {
67    e.preventDefault();
68    
69    if (!validateForm()) return;
70
71    setIsLoading(true);
72    try {
73      // Send the correct format to the API
74      await api.post('/signup/', {
75        name: formData.name,
76        email: formData.email,
77        password: formData.password,
78        confirmPassword: formData.confirmPassword
79      });
80      
81      // Then login
82      await login(formData.email, formData.password);
83      navigate('/');
84    } catch (error) {
85      const errorData = error.response?.data;
86      if (errorData) {
87        setErrors(errorData);
88      } else {
89        setErrors({ submit: error.message || 'Failed to create account' });
90      }
91    } finally {
92      setIsLoading(false);
93    }
94  };
95
96  return (
97    <div className="min-h-screen bg-[#FFFDF5] relative overflow-hidden">
98      {/* Background Pattern */}
99      <div 
100        className="absolute inset-0 opacity-10"
101        style={{
102          backgroundImage: 'radial-gradient(#000 1.5px, transparent 1.5px)',
103          backgroundSize: '20px 20px'
104        }}
105      ></div>
106      
107      {/* Floating Decorative Elements */}
108      <div className="absolute top-10 right-4 sm:top-20 sm:right-10 rotate-12">
109        <Star className="w-6 h-6 sm:w-8 sm:h-8 fill-[#FF6B6B] text-[#FF6B6B] animate-spin-slow" />
110      </div>

121          <motion.div
122            initial={{ opacity: 0, x: -50 }}
123            animate={{ opacity: 1, x: 0 }}
124            transition={{ duration: 0.5 }}
125            className="w-full max-w-md"
126          >

129              <div className="flex items-center justify-center space-x-3 mb-6">

131                  <Pill className="w-8 h-8 stroke-[3px] text-black" />

135            </div>

138            <div className="text-center mb-8">

248                    <p className="font-bold text-xs text-[#FF6B6B] mt-1 uppercase">{errors.password}</p>

258                    <input
259                      id="confirmPassword"

262                      required

264                      onChange={handleChange}

295                    initial={{ opacity: 0, y: -10 }}
296                    animate={{ opacity: 1, y: 0 }}
297                    className="bg-[#FF6B6B] border-4 border-black p-4 rotate-1"
298                  >
299                    <p className="font-bold text-sm text-white text-center uppercase">{errors.submit}</p>
300                  </motion.div>
301                )}

304                <button
305                  type="submit"
306                  disabled={isLoading}
307                  className="w-full bg-[#FFD93D] border-4 border-black font-black uppercase tracking-widest text-sm py-4 h-14
308                    shadow-[6px_6px_0px_0px_#000]
309                    hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
310                    active:shadow-none active:translate-x-[6px] active:translate-y-[6px]

312                    transition-all duration-100 -rotate-1"

315                </button>

320            <motion.div
321              initial={{ opacity: 0 }}
322              animate={{ opacity: 1 }}
323              transition={{ delay: 0.6 }}
324              className="text-center mt-8"
325            >
326              <div className="bg-[#FF6B6B] border-4 border-black p-6 shadow-[6px_6px_0px_0px_#000] rotate-1">
327                <p className="font-bold text-white uppercase tracking-wide">
328                  ALREADY HAVE AN ACCOUNT?{' '}
329                  <Link 
330                    to="/login" 
331                    className="font-black text-white underline decoration-4 underline-offset-4 hover:text-[#FFD93D] transition-colors duration-100"
332                  >
333                    SIGN IN HERE

338

340            <motion.div
341              initial={{ opacity: 0, y: 20 }}
342              animate={{ opacity: 1, y: 0 }}
343              transition={{ delay: 0.8 }}
344              className="lg:hidden mt-8"
345            >
346              <div className="bg-[#C4B5FD] border-4 border-black p-6 shadow-[6px_6px_0px_0px_#000] -rotate-1">
347                <h3 className="font-black uppercase tracking-widest text-sm text-center mb-4 rotate-1">
348                  JOIN 50,000+ HAPPY CUSTOMERS!
349                </h3>
