1import { useState, useRef, useEffect } from 'react'
2import { motion } from 'framer-motion'
3import { useNavigate } from 'react-router-dom'
4import { ChevronLeft, ChevronRight } from 'lucide-react'
5import Button from './Button'
6import Card from './Card'
7
8export default function TrendingCarousel({ products, title = "Trending Products", showTitle = true }) {
9  const [isDragging, setIsDragging] = useState(false)
10  const [startX, setStartX] = useState(0)
11  const [scrollLeft, setScrollLeft] = useState(0)
12  const carouselRef = useRef(null)
13  const navigate = useNavigate()
14
15  const handleMouseDown = (e) => {
16    setIsDragging(true)
17    setStartX(e.pageX - carouselRef.current.offsetLeft)
18    setScrollLeft(carouselRef.current.scrollLeft)
19  }
20
21  const handleMouseMove = (e) => {
22    if (!isDragging) return
23    e.preventDefault()
24    const x = e.pageX - carouselRef.current.offsetLeft
25    const walk = (x - startX) * 2
26    carouselRef.current.scrollLeft = scrollLeft - walk
27  }
28
29  const handleMouseUp = () => {
30    setIsDragging(false)
31  }
32
33  const handleTouchStart = (e) => {
34    setIsDragging(true)
35    setStartX(e.touches[0].pageX - carouselRef.current.offsetLeft)
36    setScrollLeft(carouselRef.current.scrollLeft)
37  }
38
39  const handleTouchMove = (e) => {
40    if (!isDragging) return
41    const x = e.touches[0].pageX - carouselRef.current.offsetLeft
42    const walk = (x - startX) * 2
43    carouselRef.current.scrollLeft = scrollLeft - walk
44  }
45
46  const handleTouchEnd = () => {
47    setIsDragging(false)
48  }
49
50  const scrollCarousel = (direction) => {
51    if (carouselRef.current) {
52      const scrollAmount = 300
53      carouselRef.current.scrollBy({
54        left: direction === 'left' ? -scrollAmount : scrollAmount,
55        behavior: 'smooth'
56      })
57    }
58  }
59
60  useEffect(() => {
61    const carousel = carouselRef.current
62    if (carousel) {
63      carousel.addEventListener('mousedown', handleMouseDown)
64      carousel.addEventListener('mousemove', handleMouseMove)
65      carousel.addEventListener('mouseup', handleMouseUp)
66      carousel.addEventListener('mouseleave', handleMouseUp)
67      carousel.addEventListener('touchstart', handleTouchStart)
68      carousel.addEventListener('touchmove', handleTouchMove)
69      carousel.addEventListener('touchend', handleTouchEnd)
70
71      return () => {
72        carousel.removeEventListener('mousedown', handleMouseDown)
73        carousel.removeEventListener('mousemove', handleMouseMove)
74        carousel.removeEventListener('mouseup', handleMouseUp)
75        carousel.removeEventListener('mouseleave', handleMouseUp)
76        carousel.removeEventListener('touchstart', handleTouchStart)
77        carousel.removeEventListener('touchmove', handleTouchMove)
78        carousel.removeEventListener('touchend', handleTouchEnd)
79      }
80    }
81  }, [isDragging, startX, scrollLeft])
82
83  if (!products || products.length === 0) return null
84
85  return (
86    <div className="relative">
87      {showTitle && (
88        <div className="flex justify-between items-center mb-6">
89          <h2 className="text-2xl md:text-3xl font-black uppercase -rotate-1">{title}</h2>
90          <div className="flex gap-2">
91            <Button 
92              variant="ghost" 
93              size="sm" 
94              className="rotate-1"
95              onClick={() => scrollCarousel('left')}
96            >
97              <ChevronLeft className="w-6 h-6 stroke-[3px]" />
98            </Button>
99            <Button 
100              variant="ghost" 
101              size="sm" 
102              className="-rotate-1"
103              onClick={() => scrollCarousel('right')}
104            >
105              <ChevronRight className="w-6 h-6 stroke-[3px]" />
106            </Button>
107          </div>
108        </div>
109      )}
110
111      <div 
112        ref={carouselRef}
113        className="flex gap-4 overflow-x-auto scrollbar-hide cursor-grab active:cursor-grabbing pb-4"
114        style={{ 
115          scrollBehavior: 'smooth',
116          userSelect: 'none'
117        }}
118      >
119        {products.map((product, index) => (
120          <motion.div
121            key={product.product_id}
122            className="flex-shrink-0 w-64"
123            initial={{ opacity: 0, x: 50 }}
124            animate={{ opacity: 1, x: 0 }}
125            transition={{ delay: index * 0.1 }}
126            onClick={() => !isDragging && navigate(`/product/${product.slug}`)}
127          >
128            <Card
129              className="overflow-hidden cursor-pointer"
130              rotation={Math.random() > 0.7 ? (Math.random() > 0.5 ? 1 : -1) : 0}
131              onMouseDown={(e) => e.preventDefault()}
132            >
133              <div className="relative">
134                <img 
135                  src={product.images[0]} 
136                  alt={product.name} 
137                  className="w-full h-40 object-cover border-b-4 border-neo-ink"
138                  loading="lazy"
139                />
140                {product.trending && (
141                  <span className="absolute -top-2 -left-2 neo-badge bg-neo-accent rotate-12 z-10">
142                    🔥 HOT
143                  </span>
144                )}
145                {product.discount > 0 && (
146                  <span className="absolute -top-2 -right-2 neo-badge bg-neo-secondary -rotate-12 z-10">
147                    {product.discount}% OFF
148                  </span>
149                )}
150              </div>
151              <div className="p-4">
152                <p className="text-sm font-bold uppercase tracking-wide">{product.company}</p>
153                <h3 className="text-base font-black mt-1 mb-3 line-clamp-2 leading-tight">
154                  {product.name}
155                </h3>
156                <div className="flex justify-between items-center mb-4">
157                  <div>
158                    <p className="text-xl font-black">₹{product.discounted_price.toFixed(2)}</p>
159                    {product.discount > 0 && (
160                      <p className="text-sm font-bold line-through opacity-60">₹{product.mrp.toFixed(2)}</p>
161                    )}
162                  </div>
163                  <span className="neo-badge bg-neo-secondary rotate-3">
164                    {product.discount}% OFF
165                  </span>
166                </div>
167                <Button variant="primary" size="sm" className="w-full">
168                  ADD TO CART
169                </Button>
170              </div>
171            </Card>
172          </motion.div>
173        ))}
174      </div>
175    </div>
176  )
177}
