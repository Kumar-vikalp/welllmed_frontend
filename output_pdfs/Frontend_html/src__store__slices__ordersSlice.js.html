1import { createSlice, createAsyncThunk } from '@reduxjs/toolkit'
2import api from '../../api/axiosConfig'
3
4// Async thunks
5export const fetchOrders = createAsyncThunk(
6  'orders/fetchOrders',
7  async (_, { rejectWithValue }) => {
8    try {
9      const response = await api.get('/orders/')
10      const ordersData = Array.isArray(response.data) ? response.data : [response.data]
11      
12      return ordersData.map(order => ({
13        orderId: order.order_id,
14        date: order.created_at,
15        total: parseFloat(order.total),
16        status: order.status,
17        paymentMethod: order.payment_method,
18        estimatedDelivery: order.estimated_delivery,
19        items: order.items || [],
20        canCancel: order.can_cancel,
21        deliveryDetails: {
22          name: order.delivery_name,
23          address: order.delivery_address,
24          phone: order.delivery_phone,
25          estimatedDelivery: order.estimated_delivery
26        }
27      }))
28    } catch (error) {
29      return rejectWithValue(error.response?.data || 'Failed to fetch orders')
30    }
31  }
32)
33
34export const createOrder = createAsyncThunk(
35  'orders/createOrder',
36  async (orderData, { rejectWithValue }) => {
37    try {
38      const response = await api.post('/orders/create/', orderData)
39      return response.data
40    } catch (error) {
41      return rejectWithValue(error.response?.data || 'Failed to create order')
42    }
43  }
44)
45
46const ordersSlice = createSlice({
47  name: 'orders',
48  initialState: {
49    items: [],
50    loading: false,
51    creating: false,
52    error: null
53  },
54  reducers: {
55    clearOrders: (state) => {
56      state.items = []
57    }
58  },
59  extraReducers: (builder) => {
60    builder
61      .addCase(fetchOrders.pending, (state) => {
62        state.loading = true
63        state.error = null
64      })
65      .addCase(fetchOrders.fulfilled, (state, action) => {
66        state.loading = false
67        state.items = action.payload
68      })
69      .addCase(fetchOrders.rejected, (state, action) => {
70        state.loading = false
71        state.error = action.payload
72      })
73      .addCase(createOrder.pending, (state) => {
74        state.creating = true
75        state.error = null
76      })
77      .addCase(createOrder.fulfilled, (state, action) => {
78        state.creating = false
79        // Add new order to the beginning of the list
80        state.items.unshift(action.payload)
81      })
82      .addCase(createOrder.rejected, (state, action) => {
83        state.creating = false
84        state.error = action.payload
85      })
86  }
87})
88
89export const { clearOrders } = ordersSlice.actions
90
91// Selectors
92export const selectOrders = (state) => state.orders.items
93export const selectOrdersLoading = (state) => state.orders.loading
94export const selectOrderCreating = (state) => state.orders.creating
95export const selectOrdersError = (state) => state.orders.error
96
97export default ordersSlice.reducer
