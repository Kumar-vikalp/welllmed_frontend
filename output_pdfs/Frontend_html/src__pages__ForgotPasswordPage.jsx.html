1import { useState } from 'react';
2import { Link, useNavigate } from 'react-router-dom';
3import { motion } from 'framer-motion';
4import { useUser } from '../contexts/UserContext';
5import Toast from '../components/Toast';
6
7export default function ForgotPasswordPage() {
8  const [email, setEmail] = useState('');
9  const [loading, setLoading] = useState(false);
10  const [toast, setToast] = useState({ message: '', type: 'info' });
11  const { forgotPassword } = useUser();
12  const navigate = useNavigate();
13
14  const handleSubmit = async (e) => {
15    e.preventDefault();
16    setLoading(true);
17    setToast({ message: '', type: 'info' });
18
19    try {
20      const response = await forgotPassword(email);
21      setToast({ message: response.message || 'OTP sent successfully! Redirecting...', type: 'success' });
22      
23      setTimeout(() => {
24        navigate('/reset-password');
25      }, 2000);
26
27    } catch (err) {
28      const errorMessage = err.response?.data?.detail || 'An error occurred. Please try again.';
29      setToast({ message: errorMessage, type: 'error' });
30      setLoading(false);
31    }
32  };
33
34  return (
35    <div className="min-h-screen bg-[#FFFDF5] flex items-center justify-center p-4 relative overflow-hidden">
36      <Toast message={toast.message} type={toast.type} duration={5000} />
37      
38      {/* Background Pattern */}
39      <div 
40        className="absolute inset-0 opacity-20"
41        style={{
42          backgroundImage: 'radial-gradient(#000 1.5px, transparent 1.5px)',
43          backgroundSize: '20px 20px'
44        }}
45      ></div>
46      
47      {/* Floating Decorative Elements */}
48      <div className="absolute top-10 left-4 sm:top-20 sm:left-10 rotate-12">
49        <div className="w-6 h-6 sm:w-8 sm:h-8 bg-[#FF6B6B] border-2 border-black animate-spin-slow"></div>
50      </div>
51      <div className="absolute top-32 right-4 sm:top-40 sm:right-20 -rotate-12">
52        <div className="w-8 h-8 sm:w-12 sm:h-12 bg-[#FFD93D] border-2 border-black animate-bounce"></div>
53      </div>
54      <div className="absolute bottom-20 left-8 sm:bottom-32 sm:left-32 rotate-45">
55        <div className="w-6 h-6 sm:w-10 sm:h-10 bg-[#C4B5FD] border-2 border-black"></div>
56      </div>
57
58      <motion.div 
59        initial={{ opacity: 0, y: -20 }}
60        animate={{ opacity: 1, y: 0 }}
61        className="w-full max-w-md relative z-10"
62      >
63        <div className="text-center mb-8">
64          <Link to="/" className="inline-block mb-6">
65            <div className="bg-[#FFD93D] border-4 border-black p-4 shadow-[6px_6px_0px_0px_#000] -rotate-1">
66              <span className="text-3xl sm:text-4xl font-black uppercase tracking-tight">
67                genx
68              </span>
69            </div>
70          </Link>
71          <h2 className="text-2xl sm:text-3xl font-black uppercase tracking-tight mb-4 rotate-1">
72            FORGOT PASSWORD?
73          </h2>
74          <p className="font-bold uppercase tracking-wide text-sm sm:text-base">
75            NO PROBLEM. ENTER YOUR EMAIL AND WE'LL SEND YOU AN OTP TO RESET IT.
76          </p>
77        </div>
78
79        <div className="bg-white border-4 border-black shadow-[12px_12px_0px_0px_#000] p-6 sm:p-8 -rotate-1">
80          <form onSubmit={handleSubmit} className="space-y-6">
81            <div>
82              <label htmlFor="email" className="block text-xs font-black uppercase tracking-widest mb-2">
83                EMAIL ADDRESS *
84              </label>
85              <input
86                id="email"
87                name="email"
88                type="email"
89                autoComplete="email"
90                required
91                value={email}
92                onChange={(e) => setEmail(e.target.value)}
93                className="w-full border-4 border-black bg-white font-bold text-base h-14 px-4
94                  placeholder:text-black/40 placeholder:font-bold placeholder:uppercase
95                  focus-visible:bg-[#FFD93D] focus-visible:shadow-[4px_4px_0px_0px_#000]
96                  focus-visible:outline-none focus-visible:ring-0
97                  transition-all duration-100"
98                placeholder="ENTER YOUR EMAIL"
99              />
100            </div>
101            
102            <button
103              type="submit"
104              disabled={loading}
105              className="w-full bg-[#FFD93D] border-4 border-black font-black uppercase tracking-widest 
106                text-sm py-4 h-14 disabled:opacity-50 disabled:cursor-not-allowed
107                shadow-[6px_6px_0px_0px_#000]
108                hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
109                active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
110                transition-all duration-100"
111            >
112              {loading ? (
113                <div className="flex items-center justify-center">
114                  <div className="w-4 h-4 border-2 border-black mr-2 animate-spin"></div>
115                  SENDING OTP...
116                </div>
117              ) : (
118                'SEND RESET OTP'
119              )}
120            </button>
121          </form>
122        </div>
123
124        <div className="text-center mt-8">
125          <div className="bg-[#C4B5FD] border-4 border-black p-4 shadow-[6px_6px_0px_0px_#000] rotate-1">
126            <p className="font-bold uppercase tracking-wide text-sm">
127              REMEMBERED YOUR PASSWORD?{' '}
128              <Link 
129                to="/login" 
130                className="font-black underline decoration-4 underline-offset-4 hover:text-[#FF6B6B] transition-colors"
131              >
132                BACK TO LOGIN
133              </Link>
134            </p>
135          </div>
136        </div>
137      </motion.div>
138    </div>
139  );
140}
