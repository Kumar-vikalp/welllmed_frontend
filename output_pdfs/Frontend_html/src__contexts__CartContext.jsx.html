1import { createContext, useContext, useState, useEffect } from 'react';
2import api from '../api/axiosConfig';
3import { useUser } from './UserContext';
4
5const CartContext = createContext();
6
7export function CartProvider({ children }) {
8  const [cart, setCart] = useState([]);
9  const [loading, setLoading] = useState(false);
10  const { user } = useUser();
11
12  // Fetch cart from API when user is available
13  useEffect(() => {
14    if (user) {
15      fetchCart();
16    } else {
17      setCart([]);
18    }
19  }, [user]);
20
21  const fetchCart = async () => {
22    if (!user) return;
23    
24    try {
25      setLoading(true);
26      const response = await api.get('/cart/');
27      // Transform API response to match our cart structure
28      const cartItems = Array.isArray(response.data) ? response.data : [response.data];
29      const transformedCart = cartItems.map(item => ({
30        product_id: item.product.product_id,
31        cart_item_id: item.cart_item_id,
32        name: item.product.name,
33        company: item.product.company,
34        disease_category: item.product.disease_category,
35        mrp: parseFloat(item.product.mrp),
36        discount: item.product.discount,
37        images: item.product.images?.map(img => img.stream_url) || ['/images/placeholder.jpg'],
38        trending: item.product.trending,
39        slug: item.product.slug,
40        qty: item.quantity,
41        available_stock: item.product.available_stock || 0,
42        discounted_price: parseFloat(item.product.discounted_price)
43      }));
44      setCart(transformedCart);
45    } catch (error) {
46      console.error('Failed to fetch cart', error);
47      setCart([]);
48    } finally {
49      setLoading(false);
50    }
51  };
52
53  const addToCart = async (product, qty = 1) => {
54    if (!user) {
55      throw new Error('Please login to add items to cart');
56    }
57
58    try {
59      await api.post('/cart/add/', {
60        product_id: product.product_id,
61        quantity: qty
62      });
63      
64      // Refresh cart after adding
65      await fetchCart();
66    } catch (error) {
67      console.error('Failed to add item to cart', error);
68      throw error;
69    }
70  };
71
72  const removeFromCart = async (productId) => {
73    if (!user) return;
74
75    try {
76      const cartItem = cart.find(item => item.product_id === productId);
77      if (cartItem && cartItem.cart_item_id) {
78        await api.delete(`/cart/remove/${cartItem.cart_item_id}/`);
79        await fetchCart();
80      }
81    } catch (error) {
82      console.error('Failed to remove item from cart', error);
83      throw error;
84    }
85  };
86
87  const updateQty = async (productId, qty) => {
88    if (!user) return;
89
90    if (qty < 1) {
91      await removeFromCart(productId);
92      return;
93    }
94
95    try {
96      const cartItem = cart.find(item => item.product_id === productId);
97      if (cartItem && cartItem.cart_item_id) {
98        await api.put(`/cart/update/${cartItem.cart_item_id}/`, {
99          quantity: qty
100        });
101        
102        await fetchCart();
103      }
104    } catch (error) {
105      console.error('Failed to update cart item quantity', error);
106      throw error;
107    }
108  };
109
110  const clearCart = () => {
111    setCart([]);
112  };
113
114  const cartCount = cart.reduce((count, item) => count + item.qty, 0);
115  
116  const cartTotal = cart.reduce((total, item) => {
117    const price = item.discounted_price || (item.mrp - (item.mrp * item.discount / 100));
118    return total + (price * item.qty);
119  }, 0);
120
121  const value = {
122    cart,
123    loading,
124    addToCart,
125    removeFromCart,
126    updateQty,
127    clearCart,
128    fetchCart,
129    cartCount,
130    cartTotal,
131  };
132
133  return (
134    <CartContext.Provider value={value}>
135      {children}
136    </CartContext.Provider>
137  );
138}
139
140export const useCart = () => useContext(CartContext);
