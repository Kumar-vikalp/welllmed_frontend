1import React from "react";
2
3export default function SEO({
4  title,
5  description,
6  name,
7  type,
8  keywords,
9  link,
10}) {
11  React.useEffect(() => {
12    // Update document title
13    if (title) {
14      document.title = title;
15    }
16    
17    // Update meta description
18    if (description) {
19      let metaDescription = document.querySelector('meta[name="description"]');
20      if (!metaDescription) {
21        metaDescription = document.createElement('meta');
22        metaDescription.name = 'description';
23        document.head.appendChild(metaDescription);
24      }
25      metaDescription.content = description;
26    }
27    
28    // Update meta keywords
29    if (keywords) {
30      let metaKeywords = document.querySelector('meta[name="keywords"]');
31      if (!metaKeywords) {
32        metaKeywords = document.createElement('meta');
33        metaKeywords.name = 'keywords';
34        document.head.appendChild(metaKeywords);
35      }
36      metaKeywords.content = keywords;
37    }
38    
39    // Update canonical link
40    if (link) {
41      let canonicalLink = document.querySelector('link[rel="canonical"]');
42      if (!canonicalLink) {
43        canonicalLink = document.createElement('link');
44        canonicalLink.rel = 'canonical';
45        document.head.appendChild(canonicalLink);
46      }
47      canonicalLink.href = link;
48    }
49    
50    // Update Open Graph tags
51    if (title) {
52      let ogTitle = document.querySelector('meta[property="og:title"]');
53      if (!ogTitle) {
54        ogTitle = document.createElement('meta');
55        ogTitle.setAttribute('property', 'og:title');
56        document.head.appendChild(ogTitle);
57      }
58      ogTitle.content = title;
59    }
60    
61    if (description) {
62      let ogDescription = document.querySelector('meta[property="og:description"]');
63      if (!ogDescription) {
64        ogDescription = document.createElement('meta');
65        ogDescription.setAttribute('property', 'og:description');
66        document.head.appendChild(ogDescription);
67      }
68      ogDescription.content = description;
69    }
70    
71    if (type) {
72      let ogType = document.querySelector('meta[property="og:type"]');
73      if (!ogType) {
74        ogType = document.createElement('meta');
75        ogType.setAttribute('property', 'og:type');
76        document.head.appendChild(ogType);
77      }
78      ogType.content = type;
79    }
80    
81    // Update Twitter Card tags
82    if (title) {
83      let twitterTitle = document.querySelector('meta[name="twitter:title"]');
84      if (!twitterTitle) {
85        twitterTitle = document.createElement('meta');
86        twitterTitle.name = 'twitter:title';
87        document.head.appendChild(twitterTitle);
88      }
89      twitterTitle.content = title;
90    }
91    
92    if (description) {
93      let twitterDescription = document.querySelector('meta[name="twitter:description"]');
94      if (!twitterDescription) {
95        twitterDescription = document.createElement('meta');
96        twitterDescription.name = 'twitter:description';
97        document.head.appendChild(twitterDescription);
98      }
99      twitterDescription.content = description;
100    }
101    
102    if (type) {
103      let twitterCard = document.querySelector('meta[name="twitter:card"]');
104      if (!twitterCard) {
105        twitterCard = document.createElement('meta');
106        twitterCard.name = 'twitter:card';
107        document.head.appendChild(twitterCard);
108      }
109      twitterCard.content = type;
110    }
111    
112    if (name) {
113      let twitterCreator = document.querySelector('meta[name="twitter:creator"]');
114      if (!twitterCreator) {
115        twitterCreator = document.createElement('meta');
116        twitterCreator.name = 'twitter:creator';
117        document.head.appendChild(twitterCreator);
118      }
119      twitterCreator.content = name;
120    }
121  }, [title, description, name, type, keywords, link]);
122  
123  return (
124    null
125  );
126}
