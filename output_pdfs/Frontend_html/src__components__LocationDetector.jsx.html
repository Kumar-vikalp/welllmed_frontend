1import { useState, useEffect } from 'react'
2import { detectUserLocation } from '../utils/locationUtils'
3
4export default function LocationDetector({ onLocationDetected }) {
5  const [pincode, setPincode] = useState('') // Start empty
6  const [detecting, setDetecting] = useState(false)
7  const [showModal, setShowModal] = useState(false)
8
9  useEffect(() => {
10    const savedPincode = localStorage.getItem('user_pincode')
11    if (savedPincode) {
12      setPincode(savedPincode)
13      onLocationDetected?.(savedPincode)
14    } else {
15      // Set default pincode if none saved
16      setPincode('110040')
17      localStorage.setItem('user_pincode', '110040')
18      onLocationDetected?.('110040')
19    }
20  }, [onLocationDetected])
21
22  const handleDetectLocation = async () => {
23    setDetecting(true)
24    try {
25      const locationData = await detectUserLocation()
26      if (locationData.pincode) {
27        setPincode(locationData.pincode)
28        localStorage.setItem('user_pincode', locationData.pincode)
29        onLocationDetected?.(locationData.pincode)
30        setShowModal(false)
31      }
32    } catch (error) {
33      console.log('Could not detect location, using default pincode')
34      // Keep current pincode on error
35    } finally {
36      setDetecting(false)
37    }
38  }
39
40  const handlePincodeSubmit = () => {
41    if (pincode && pincode.length === 6) {
42      localStorage.setItem('user_pincode', pincode)
43      onLocationDetected?.(pincode)
44      setShowModal(false)
45    }
46  }
47
48  const handleInputChange = (e) => {
49    const value = e.target.value.replace(/\D/g, '').slice(0, 6) // Only numbers, max 6 digits
50    setPincode(value)
51  }
52
53  const handleCancel = () => {
54    // Reset to saved pincode or default
55    const savedPincode = localStorage.getItem('user_pincode') || '110040'
56    setPincode(savedPincode)
57    setShowModal(false)
58  }
59
60  return (
61    <>
62      <div className="flex items-center gap-2 text-xs md:text-sm">
63        <svg className="w-3 h-3 md:w-4 md:h-4 text-purple-500" fill="none" stroke="currentColor" viewBox="0 0 24 24">
64          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M17.657 16.657L13.414 20.9a1.998 1.998 0 01-2.827 0l-4.244-4.243a8 8 0 1111.314 0z" />
65          <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M15 11a3 3 0 11-6 0 3 3 0 016 0z" />
66        </svg>
67        <span className="text-gray-600">Deliver to:</span>
68        <button 
69          onClick={() => setShowModal(true)}
70          className="text-purple-600 hover:text-purple-700 font-medium"
71        >
72          {pincode || 'Select'}
73        </button>
74        {detecting && <span className="text-xs text-gray-500">Detecting...</span>}
75      </div>
76
77      {/* Pincode Change Modal */}
78      {showModal && (
79        <div className="fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50 p-4">
80          <div className="bg-white rounded-lg p-6 max-w-md w-full">
81            <h3 className="text-xl font-bold mb-4 text-gray-900">Change Delivery Location</h3>
82            <div className="space-y-4">
83              <div>
84                <label className="block text-sm font-medium text-gray-700 mb-2">Enter Pincode</label>
85                <input
86                  type="text"
87                  value={pincode}
88                  onChange={handleInputChange}
89                  className="w-full bg-white border border-gray-300 rounded-md px-3 py-2 text-gray-900 focus:ring-2 focus:ring-primary-500 focus:border-transparent"
90                  placeholder="Enter 6-digit pincode"
91                  maxLength="6"
92                />
93              </div>
94              <div className="flex gap-3">
95                <button
96                  onClick={handlePincodeSubmit}
97                  disabled={!pincode || pincode.length !== 6}
98                  className="flex-1 bg-purple-500 hover:bg-purple-600 text-white font-medium py-2 px-4 rounded-lg disabled:bg-gray-300 disabled:cursor-not-allowed"
99                >
100                  Update
101                </button>
102                <button
103                  onClick={handleDetectLocation}
104                  disabled={detecting}
105                  className="flex-1 bg-gray-500 hover:bg-gray-600 text-white font-medium py-2 px-4 rounded-lg disabled:opacity-50"
106                >
107                  {detecting ? 'Detecting...' : 'Auto Detect'}
108                </button>
109                <button
110                  onClick={handleCancel}
111                  className="flex-1 bg-gray-200 hover:bg-gray-300 text-gray-700 font-medium py-2 px-4 rounded-lg"
112                >
113                  Cancel
114                </button>
115              </div>
116            </div>
117          </div>
118        </div>
119      )}
120    </>
121  )
122}
