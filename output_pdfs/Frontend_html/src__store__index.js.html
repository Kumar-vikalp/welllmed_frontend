src/store/index.jsJS
1import { configureStore } from '@reduxjs/toolkit'
2import { persistStore, persistReducer, FLUSH, REHYDRATE, PAUSE, PERSIST, PURGE, REGISTER } from 'redux-persist'
3import storage from 'redux-persist/lib/storage'
4import { combineReducers } from '@reduxjs/toolkit'
5
6import cartSlice from './slices/cartSlice'
7import productsSlice from './slices/productsSlice'
8import userSlice from './slices/userSlice'
9import ordersSlice from './slices/ordersSlice'
10
11const persistConfig = {
12  key: 'root',
13  storage,
14  whitelist: ['cart', 'user'], // Only persist cart and user data
15  blacklist: ['products', 'orders'] // Don't persist products and orders as they should be fresh
16}
17
18const rootReducer = combineReducers({
19  cart: cartSlice,
20  products: productsSlice,
21  user: userSlice,
22  orders: ordersSlice
23})
24
25const persistedReducer = persistReducer(persistConfig, rootReducer)
26
27export const store = configureStore({
28  reducer: persistedReducer,
29  middleware: (getDefaultMiddleware) =>
30    getDefaultMiddleware({
31      serializableCheck: {
32        ignoredActions: [FLUSH, REHYDRATE, PAUSE, PERSIST, PURGE, REGISTER]
33      }
34    }),
35  devTools: process.env.NODE_ENV !== 'production'
36})
37
38export const persistor = persistStore(store)
