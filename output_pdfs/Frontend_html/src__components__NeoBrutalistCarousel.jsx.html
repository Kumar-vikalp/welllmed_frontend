1import { useRef, useState, useEffect } from 'react'
2import { motion } from 'framer-motion'
3import { ChevronLeft, ChevronRight } from 'lucide-react'
4import Button from './Button'
5
6export default function NeoBrutalistCarousel({ 
7  children, 
8  title, 
9  showControls = true, 
10  showTitle = true,
11  className = "",
12  itemWidth = "w-64",
13  gap = "gap-4"
14}) {
15  const carouselRef = useRef(null)
16  const [canScrollLeft, setCanScrollLeft] = useState(false)
17  const [canScrollRight, setCanScrollRight] = useState(true)
18  const [isDragging, setIsDragging] = useState(false)
19  const [startX, setStartX] = useState(0)
20  const [scrollLeft, setScrollLeft] = useState(0)
21
22  // Check scroll position to show/hide controls
23  const checkScrollPosition = () => {
24    if (carouselRef.current) {
25      const { scrollLeft, scrollWidth, clientWidth } = carouselRef.current
26      setCanScrollLeft(scrollLeft > 0)
27      setCanScrollRight(scrollLeft < scrollWidth - clientWidth - 10)
28    }
29  }
30
31  useEffect(() => {
32    const carousel = carouselRef.current
33    if (carousel) {
34      carousel.addEventListener('scroll', checkScrollPosition)
35      checkScrollPosition() // Initial check
36      
37      return () => carousel.removeEventListener('scroll', checkScrollPosition)
38    }
39  }, [children])
40
41  // Mouse drag handlers
42  const handleMouseDown = (e) => {
43    setIsDragging(true)
44    setStartX(e.pageX - carouselRef.current.offsetLeft)
45    setScrollLeft(carouselRef.current.scrollLeft)
46    carouselRef.current.style.cursor = 'grabbing'
47  }
48
49  const handleMouseMove = (e) => {
50    if (!isDragging) return
51    e.preventDefault()
52    const x = e.pageX - carouselRef.current.offsetLeft
53    const walk = (x - startX) * 2
54    carouselRef.current.scrollLeft = scrollLeft - walk
55  }
56
57  const handleMouseUp = () => {
58    setIsDragging(false)
59    if (carouselRef.current) {
60      carouselRef.current.style.cursor = 'grab'
61    }
62  }
63
64  // Touch handlers
65  const handleTouchStart = (e) => {
66    setIsDragging(true)
67    setStartX(e.touches[0].pageX - carouselRef.current.offsetLeft)
68    setScrollLeft(carouselRef.current.scrollLeft)
69  }
70
71  const handleTouchMove = (e) => {
72    if (!isDragging) return
73    const x = e.touches[0].pageX - carouselRef.current.offsetLeft
74    const walk = (x - startX) * 2
75    carouselRef.current.scrollLeft = scrollLeft - walk
76  }
77
78  const handleTouchEnd = () => {
79    setIsDragging(false)
80  }
81
82  // Scroll controls
83  const scrollCarousel = (direction) => {
84    if (carouselRef.current) {
85      const scrollAmount = 300
86      carouselRef.current.scrollBy({
87        left: direction === 'left' ? -scrollAmount : scrollAmount,
88        behavior: 'smooth'
89      })
90    }
91  }
92
93  // Add event listeners
94  useEffect(() => {
95    const carousel = carouselRef.current
96    if (carousel) {
97      carousel.addEventListener('mousedown', handleMouseDown)
98      carousel.addEventListener('mousemove', handleMouseMove)
99      carousel.addEventListener('mouseup', handleMouseUp)
100      carousel.addEventListener('mouseleave', handleMouseUp)
101      carousel.addEventListener('touchstart', handleTouchStart)
102      carousel.addEventListener('touchmove', handleTouchMove)
103      carousel.addEventListener('touchend', handleTouchEnd)
104
105      return () => {
106        carousel.removeEventListener('mousedown', handleMouseDown)
107        carousel.removeEventListener('mousemove', handleMouseMove)
108        carousel.removeEventListener('mouseup', handleMouseUp)
109        carousel.removeEventListener('mouseleave', handleMouseUp)
110        carousel.removeEventListener('touchstart', handleTouchStart)
111        carousel.removeEventListener('touchmove', handleTouchMove)
112        carousel.removeEventListener('touchend', handleTouchEnd)
113      }
114    }
115  }, [isDragging, startX, scrollLeft])
116
117  return (
118    <div className={`relative ${className}`}>
119      {/* Title and Controls */}
120      {(showTitle || showControls) && (
121        <div className="flex justify-between items-center mb-6">
122          {showTitle && title && (
123            <h2 className="text-2xl md:text-3xl font-black uppercase -rotate-1">
124              {title}
125            </h2>
126          )}
127          
128          {showControls && (
129            <div className="flex gap-2">
130              <Button 
131                variant="ghost" 
132                size="sm" 
133                className={`rotate-1 ${!canScrollLeft ? 'opacity-50 cursor-not-allowed' : ''}`}
134                onClick={() => scrollCarousel('left')}
135                disabled={!canScrollLeft}
136              >
137                <ChevronLeft className="w-6 h-6 stroke-[3px]" />
138              </Button>
139              <Button 
140                variant="ghost" 
141                size="sm" 
142                className={`-rotate-1 ${!canScrollRight ? 'opacity-50 cursor-not-allowed' : ''}`}
143                onClick={() => scrollCarousel('right')}
144                disabled={!canScrollRight}
145              >
146                <ChevronRight className="w-6 h-6 stroke-[3px]" />
147              </Button>
148            </div>
149          )}
150        </div>
151      )}
152
153      {/* Carousel Container */}
154      <div className="relative">
155        <div 
156          ref={carouselRef}
157          className={`flex ${gap} overflow-x-auto scrollbar-hide cursor-grab active:cursor-grabbing pb-4 touch-pan-x`}
158          style={{ 
159            scrollBehavior: 'smooth',
160            userSelect: 'none'
161          }}
162        >
163          {children}
164        </div>
165
166        {/* Gradient Overlays for Visual Cues */}
167        {canScrollLeft && (
168          <div className="absolute left-0 top-0 bottom-4 w-8 bg-gradient-to-r from-neo-canvas to-transparent pointer-events-none z-10" />
169        )}
170        {canScrollRight && (
171          <div className="absolute right-0 top-0 bottom-4 w-8 bg-gradient-to-l from-neo-canvas to-transparent pointer-events-none z-10" />
172        )}
173      </div>
174
175      {/* Mobile Scroll Indicator */}
176      <div className="flex justify-center mt-4 md:hidden">
177        <div className="flex gap-2">
178          {Array.from({ length: Math.ceil(children?.length / 2) || 0 }).map((_, index) => (
179            <div
180              key={index}
181              className="w-2 h-2 bg-neo-ink border border-neo-ink opacity-30"
182            />
183          ))}
184        </div>
185      </div>
186    </div>
187  )
188}
