1import { createSlice, createAsyncThunk } from '@reduxjs/toolkit'
2import api from '../../api/axiosConfig'

4// Cache duration: 5 minutes
5const CACHE_DURATION = 5 * 60 * 1000
6
7// Async thunks
8export const fetchProducts = createAsyncThunk(
9  'products/fetchProducts',
10  async (params = {}, { getState, rejectWithValue }) => {
11    try {
12      const cacheKey = JSON.stringify(params)
13      const { products } = getState()
14      const cachedData = products.cache[cacheKey]
15      
16      // Return cached data if it's still valid
17      if (cachedData && Date.now() - cachedData.timestamp < CACHE_DURATION) {
18        return { products: cachedData.data, fromCache: true, cacheKey }
19      }
20      
21      const queryParams = new URLSearchParams()
22      Object.entries(params).forEach(([key, value]) => {
23        if (value) queryParams.append(key, value)
24      })
25      
26      const response = await api.get(`/products/?${queryParams.toString()}`)
27      const data = response.data
28      const productsData = data.results || (Array.isArray(data) ? data : [data])
29      
30      const transformedProducts = productsData.map(transformProduct)
31      
32      return {
33        products: transformedProducts,
34        totalPages: data.count ? Math.ceil(data.count / (params.page_size || 12)) : 1,
35        fromCache: false,
36        cacheKey
37      }
38    } catch (error) {
39      return rejectWithValue(error.response?.data || 'Failed to fetch products')
40    }
41  }
42)
43
44export const fetchProductBySlug = createAsyncThunk(
45  'products/fetchProductBySlug',
46  async (slug, { getState, rejectWithValue }) => {
47    try {
48      const { products } = getState()
49      const cachedProduct = products.productCache[slug]
50      
51      // Return cached product if it's still valid
52      if (cachedProduct && Date.now() - cachedProduct.timestamp < CACHE_DURATION) {
53        return { product: cachedProduct.data, fromCache: true }
54      }
55      
56      const response = await api.get(`/products/${slug}/`)
57      const transformedProduct = transformProduct(response.data)
58      
59      return { product: transformedProduct, fromCache: false, slug }
60    } catch (error) {
61      console.error('Product fetch error:', error)
62      return rejectWithValue(error.response?.data || 'Failed to fetch product')
63    }
64  }
65)

67export const fetchRelatedProducts = createAsyncThunk(
68  'products/fetchRelatedProducts',
69  async (slug, { rejectWithValue }) => {
70    try {
71      const response = await api.get(`/products/${slug}/related/`)
72      const relatedData = Array.isArray(response.data) ? response.data : [response.data]
73      return relatedData.map(transformProduct)
74    } catch (error) {
75      return rejectWithValue([]) // Return empty array on error
76    }
77  }
78)

80const transformProduct = (product) => ({
81  product_id: product.product_id,
82  slug: product.slug,
83  name: product.name,
84  company: product.company,
85  disease_category: product.disease_category,
86  mrp: parseFloat(product.mrp),
87  discount: product.discount,
88  discounted_price: parseFloat(product.discounted_price),
89  images: product.images?.map(img => img.stream_url) || ['/images/placeholder.jpg'],
90  trending: product.trending,
91  available_stock: product.available_stock || 100,
92  description: product.description || `${product.name} from ${product.company}. Effective treatment for ${product.disease_category}.`,
93  returnable: product.returnable,
94  expiry_date: product.expiry_date,
95  directions_for_use: product.directions_for_use,
96  seller_information: product.seller_information,
97  manufactured_by: product.manufactured_by,
98  packed_by: product.packed_by,
99  benefits: product.benefits || [],
100  suitable_for: product.suitable_for || [],
101  dosage: product.dosage || [],
102  cautions: product.cautions || [],
103  side_effects: product.side_effects || [],
104  key_ingredients: product.key_ingredients || []
105})

107const productsSlice = createSlice({
108  name: 'products',
109  initialState: {
110    items: [],
111    currentProduct: null,
112    relatedProducts: [],
113    loading: false,
114    productLoading: false,
115    error: null,
116    totalPages: 1,
117    cache: {}, // Cache for product lists
118    productCache: {}, // Cache for individual products
119    filters: {
120      category: '',
121      company: '',
122      search: '',
123      ordering: '',
124      trending: ''
125    }
126  },
127  reducers: {
128    setFilters: (state, action) => {
129      state.filters = { ...state.filters, ...action.payload }
130    },
131    clearFilters: (state) => {
132      state.filters = {
133        category: '',
134        company: '',
135        search: '',
136        ordering: '',
137        trending: ''
138      }
139    },
140    clearCurrentProduct: (state) => {
141      state.currentProduct = null
142      state.relatedProducts = []
143    },
144    clearCache: (state) => {
145      state.cache = {}
146      state.productCache = {}
147    }
148  },
149  extraReducers: (builder) => {
150    builder
151      // Fetch products
152      .addCase(fetchProducts.pending, (state) => {
153        state.loading = true
154        state.error = null
155      })
156      .addCase(fetchProducts.fulfilled, (state, action) => {
157        state.loading = false
158        state.items = action.payload.products
159        state.totalPages = action.payload.totalPages || 1
160        
161        // Cache the results if not from cache
162        if (!action.payload.fromCache) {
163          state.cache[action.payload.cacheKey] = {
164            data: action.payload.products,
165            timestamp: Date.now()
166          }
167        }
168      })
169      .addCase(fetchProducts.rejected, (state, action) => {
170        state.loading = false
171        state.error = action.payload
172      })
173      
174      // Fetch product by slug
175      .addCase(fetchProductBySlug.pending, (state) => {
176        state.productLoading = true
177        state.error = null
178      })
179      .addCase(fetchProductBySlug.fulfilled, (state, action) => {
180        state.productLoading = false
181        state.currentProduct = action.payload.product
182        
183        // Cache the product if not from cache
184        if (!action.payload.fromCache) {
185          state.productCache[action.payload.slug] = {
186            data: action.payload.product,
187            timestamp: Date.now()
188          }
189        }
190      })
191      .addCase(fetchProductBySlug.rejected, (state, action) => {
192        state.productLoading = false
193        state.error = action.payload
194      })
195      
196      // Fetch related products
197      .addCase(fetchRelatedProducts.fulfilled, (state, action) => {
198        state.relatedProducts = action.payload
199      })
200  }
201})
