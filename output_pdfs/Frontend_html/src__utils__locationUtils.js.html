3export const getCurrentLocation = () => {
4  return new Promise((resolve, reject) => {
5    if (!navigator.geolocation) {
6      reject(new Error('Geolocation is not supported by this browser.'));
7      return;
8    }
9
10    navigator.geolocation.getCurrentPosition(
11      (position) => {
12        resolve({
13          latitude: position.coords.latitude,
14          longitude: position.coords.longitude,
15        });
16      },
17      (error) => {
18        reject(error);
19      },
20      {
21        enableHighAccuracy: true,
22        timeout: 10000,
23        maximumAge: 0,
24      }
25    );
26  });
27};

29export const getAddressFromCoordinates = async (latitude, longitude) => {
30  try {
31    // Using a free geocoding service (you might want to use Google Maps API in production)
32    const response = await fetch(
33      `https://api.bigdatacloud.net/data/reverse-geocode-client?latitude=${latitude}&longitude=${longitude}&localityLanguage=en`
34    );
35    
36    if (!response.ok) {
37      throw new Error('Failed to fetch address');
38    }
39    
40    const data = await response.json();
41    
42    return {
43      address_line: `${data.locality || ''} ${data.localityInfo?.administrative?.[2]?.name || ''}`.trim(),
44      city: data.city || data.locality || '',
45      state: data.principalSubdivision || '',
46      pincode: data.postcode || '',
47      full_address: data.display_name || '',
48    };
49  } catch (error) {
50    console.error('Error getting address from coordinates:', error);
51    throw error;
52  }
53};

55export const detectUserLocation = async () => {
56  try {
57    const coordinates = await getCurrentLocation();
58    const address = await getAddressFromCoordinates(coordinates.latitude, coordinates.longitude);
59    
60    return {
61      ...address,
62      latitude: coordinates.latitude,
63      longitude: coordinates.longitude,
64    };
65  } catch (error) {
66    console.error('Error detecting user location:', error);
67    throw error;
68  }
69};
