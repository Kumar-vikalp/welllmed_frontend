1import { useState } from 'react';
2import { useNavigate } from 'react-router-dom';
3import { useDispatch, useSelector } from 'react-redux';
4import { selectCartItems, selectCartTotal, clearCart } from '../store/slices/cartSlice';
5import { selectUser } from '../store/slices/userSlice';
6import { createOrder } from "../store/slices/ordersSlice";
7import { motion } from 'framer-motion';
8
9export default function CheckoutPage() {
10  const dispatch = useDispatch();
11  const user = useSelector(selectUser);
12  const cart = useSelector(selectCartItems);
13  const cartTotal = useSelector(selectCartTotal);
14  const navigate = useNavigate();
15
16  const [paymentMethod, setPaymentMethod] = useState('UPI');
17  const [isProcessing, setIsProcessing] = useState(false);
18
19  const formatPrice = (price) => {
20    return new Intl.NumberFormat('en-IN', {
21      style: 'currency',
22      currency: 'INR',
23    }).format(price);
24  };
25
26  const handlePlaceOrder = async (e) => {
27    e.preventDefault();
28    setIsProcessing(true);
29
30    try {
31      const orderItems = cart.map((item) => ({
32        product: item.product_id,
33        quantity: item.qty,
34      }));
35
36      const orderData = {
37        paymentMethod: paymentMethod,
38        items: orderItems,
39      };
40
41      const resultAction = await dispatch(createOrder(orderData));
42      const result = resultAction.payload;
43
44      if (result) {
45        dispatch(clearCart());
46        navigate('/order-success', {
47          state: {
48            orderId: result.order_id || `ORD-${Date.now()}`,
49          },
50        });
51      }
52    } catch (error) {
53      console.error('Failed to create order', error);
54    } finally {
55      setIsProcessing(false);
56    }
57  };
58
59  if (cart.length === 0) {
60    return (
61      <div className="min-h-screen bg-[#FFFDF5] p-4 sm:p-8">
62        <div className="max-w-7xl mx-auto text-center py-20">
63          <h1 className="text-3xl sm:text-5xl font-black uppercase tracking-tight mb-4">YOUR CART IS EMPTY</h1>
64          <p className="text-base sm:text-lg font-bold uppercase tracking-wide mb-8">YOU CAN'T CHECKOUT WITHOUT ANY ITEMS</p>
65          <button
66            onClick={() => navigate('/products')}
67            className="bg-[#FF6B6B] border-4 border-black font-black uppercase tracking-widest 
68              text-sm px-6 py-4 h-14 w-full sm:w-auto
69              shadow-[6px_6px_0px_0px_#000]
70              hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
71              active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
72              transition-all duration-100"
73          >
74            START SHOPPING
75          </button>
76        </div>
77      </div>
78    );
79  }
80
81  return (
82    <motion.div
83      initial={{ opacity: 0 }}
84      animate={{ opacity: 1 }}
85      className="min-h-screen bg-[#FFFDF5] p-4 sm:p-8"
86    >
87      <div className="max-w-7xl mx-auto">
88        <h1 className="text-3xl sm:text-5xl md:text-7xl font-black uppercase tracking-tight text-center mb-8 sm:mb-12">
89          CHECKOUT
90        </h1>
91
92        <form onSubmit={handlePlaceOrder} className="grid grid-cols-1 lg:grid-cols-3 gap-8">
93          {/* Left Side */}
94          <div className="lg:col-span-2 space-y-8">
95            {/* Shipping Address */}
96            <div className="bg-white border-4 border-black shadow-[6px_6px_0px_0px_#000] overflow-hidden">
97              <div className="bg-[#C4B5FD] border-b-4 border-black p-4 sm:p-6">
98                <h2 className="text-xl sm:text-2xl font-black uppercase tracking-tight">SHIPPING ADDRESS</h2>
99              </div>
100              <div className="p-4 sm:p-6">
101                <div className="bg-[#FFD93D] border-4 border-black p-4 shadow-[4px_4px_0px_0px_#000]">
102                  <p className="font-black text-lg uppercase tracking-wide">{user?.name}</p>
103                  <p className="font-bold">{user?.address}</p>
104                  <p className="font-bold">{user?.phone}</p>
105                </div>
106              </div>
107            </div>
108
109            {/* Payment Method */}
110            <div className="bg-white border-4 border-black shadow-[6px_6px_0px_0px_#000] overflow-hidden">
111              <div className="bg-[#FFD93D] border-b-4 border-black p-4 sm:p-6">
112                <h2 className="text-xl sm:text-2xl font-black uppercase tracking-tight">PAYMENT METHOD</h2>
113              </div>
114              <div className="p-4 sm:p-6 space-y-4">
115                {/* UPI Option */}
116                <label
117                  className={`flex items-center p-4 sm:p-6 border-4 border-black cursor-pointer transition-all duration-100 ${paymentMethod === 'UPI'
118                      ? 'bg-black shadow-[2px_2px_0px_0px_#000] translate-x-[2px] translate-y-[2px]'
119                      : 'bg-white hover:bg-[#C4B5FD] shadow-[4px_4px_0px_0px_#000] hover:shadow-[2px_2px_0px_0px_#000] hover:translate-x-[2px] hover:translate-y-[2px]'
120                    }`}
121                >
122                  <input
123                    type="radio"
124                    name="payment"
125                    value="UPI"
126                    checked={paymentMethod === 'UPI'}
127                    onChange={() => setPaymentMethod('UPI')}
128                    className="sr-only"
129                  />
130                  {/* Radio Circle: Flips border color */}
131                  <div className={`w-6 h-6 border-4 mr-4 flex items-center justify-center ${paymentMethod === 'UPI' ? 'border-white' : 'border-black'}`}>
132                    {paymentMethod === 'UPI' && <div className="w-2 h-2 bg-white"></div>}
133                  </div>
134                  {/* Text: Flips to white when selected */}
135                  <span className={`text-lg font-black uppercase tracking-wide ${paymentMethod === 'UPI' ? 'text-white' : 'text-black'}`}>
136                    UPI PAYMENT
137                  </span>
138                </label>
139
140                {/* COD Option */}
141                <label
142                  className={`flex items-center p-4 sm:p-6 border-4 border-black cursor-pointer transition-all duration-100 ${paymentMethod === 'COD'
143                      ? 'bg-black shadow-[2px_2px_0px_0px_#000] translate-x-[2px] translate-y-[2px]'
144                      : 'bg-white hover:bg-[#C4B5FD] shadow-[4px_4px_0px_0px_#000] hover:shadow-[2px_2px_0px_0px_#000] hover:translate-x-[2px] hover:translate-y-[2px]'
145                    }`}
146                >
147                  <input
148                    type="radio"
149                    name="payment"
150                    value="COD"
151                    checked={paymentMethod === 'COD'}
152                    onChange={() => setPaymentMethod('COD')}
153                    className="sr-only"
154                  />
155                  <div className={`w-6 h-6 border-4 mr-4 flex items-center justify-center ${paymentMethod === 'COD' ? 'border-white' : 'border-black'}`}>
156                    {paymentMethod === 'COD' && <div className="w-2 h-2 bg-white"></div>}
157                  </div>
158                  {/* Text: Flips to white when selected */}
159                  <span className={`text-lg font-black uppercase tracking-wide ${paymentMethod === 'COD' ? 'text-white' : 'text-black'}`}>
160                    CASH ON DELIVERY
161                  </span>
162                </label>
163              </div>
164            </div>
165          </div>
166
167          {/* Right Side: Order Summary */}
168          <div className="lg:col-span-1">
169            <div className="bg-white border-4 border-black shadow-[8px_8px_0px_0px_#000] sticky top-24 overflow-hidden">
170              <div className="bg-[#FF6B6B] border-b-4 border-black p-4 sm:p-6">
171                <h2 className="text-xl sm:text-2xl font-black uppercase tracking-tight text-white">ORDER SUMMARY</h2>
172              </div>
173
174              <div className="p-4 sm:p-6">
175                <div className="space-y-3 mb-6">
176                  {cart.map((item) => (
177                    <div key={item.product_id} className="flex justify-between items-center text-sm border-b-2 border-black pb-2">
178                      <div className="flex-1">
179                        <span className="font-bold uppercase tracking-wide">
180                          {item.name} × {item.qty}
181                        </span>
182                      </div>
183                      <span className="font-black">
184                        {formatPrice((item.mrp - (item.mrp * item.discount) / 100) * item.qty)}
185                      </span>
186                    </div>
187                  ))}
188                </div>
189
190                <div className="space-y-4 mb-6">
191                  <div className="flex justify-between items-center pb-2 border-b-2 border-black">
192                    <span className="font-bold uppercase tracking-wide">SUBTOTAL</span>
193                    <span className="font-black text-lg">{formatPrice(cartTotal)}</span>
194                  </div>
195                  <div className="flex justify-between items-center pb-2 border-b-2 border-black">
196                    <span className="font-bold uppercase tracking-wide">SHIPPING</span>
197                    <span className="font-black bg-[#C4B5FD] border-2 border-black px-2 py-1 text-xs">FREE</span>
198                  </div>
199                  <div className="bg-black text-white p-4 border-4 border-black">
200                    <div className="flex justify-between items-center">
201                      <span className="text-white uppercase tracking-widest text-xl">TOTAL</span>
202                      <span className="text-white text-2xl">{formatPrice(cartTotal)}</span>
203                    </div>
204                  </div>
205                </div>
206
207                <button
208                  type="submit"
209                  disabled={isProcessing}
210                  className="w-full bg-[#FF6B6B] border-4 border-black font-black uppercase tracking-widest 
211                    text-sm py-4 h-14 disabled:opacity-50 disabled:cursor-not-allowed
212                    shadow-[6px_6px_0px_0px_#000]
213                    hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
214                    active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
215                    transition-all duration-100"
216                >
217                  {isProcessing ? (
218                    <div className="flex items-center justify-center">
219                      <div className="w-4 h-4 border-2 border-white mr-2 animate-spin"></div>
220                      PLACING ORDER...
221                    </div>
222                  ) : (
223                    `PLACE ORDER ${formatPrice(cartTotal)}`
224                  )}
225                </button>
226              </div>
227            </div>
228          </div>
229        </form>
230      </div>
231    </motion.div>
232  );
233}
