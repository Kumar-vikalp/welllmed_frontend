1import { useState, useRef, useEffect } from 'react'
2import { useInView } from 'react-intersection-observer'
3
4export default function LazyImage({ 
5  src, 
6  alt, 
7  className = '', 
8  placeholder = '/images/placeholder.jpg',
9  ...props 
10}) {
11  const [imageSrc, setImageSrc] = useState(placeholder)
12  const [imageRef, setImageRef] = useState()
13  const [loaded, setLoaded] = useState(false)
14  const [error, setError] = useState(false)
15  
16  const { ref, inView } = useInView({
17    threshold: 0.1,
18    triggerOnce: true
19  })
20
21  useEffect(() => {
22    if (inView && src && !loaded && !error) {
23      const img = new Image()
24      img.onload = () => {
25        setImageSrc(src)
26        setLoaded(true)
27      }
28      img.onerror = () => {
29        setError(true)
30        setImageSrc(placeholder)
31      }
32      img.src = src
33    }
34  }, [inView, src, loaded, error, placeholder])
35
36  return (
37    <div ref={ref} className={`relative overflow-hidden ${className}`}>
38      <img
39        src={imageSrc}
40        alt={alt}
41        className={`transition-opacity duration-300 ${
42          loaded ? 'opacity-100' : 'opacity-70'
43        } ${className}`}
44        {...props}
45      />
46      {!loaded && !error && (
47        <div className="absolute inset-0 bg-gray-200 animate-pulse flex items-center justify-center">
48          <svg className="w-8 h-8 text-gray-400" fill="none" stroke="currentColor" viewBox="0 0 24 24">
49            <path strokeLinecap="round" strokeLinejoin="round" strokeWidth="2" d="M4 16l4.586-4.586a2 2 0 012.828 0L16 16m-2-2l1.586-1.586a2 2 0 012.828 0L20 14m-6-6h.01M6 20h12a2 2 0 002-2V6a2 2 0 00-2-2H6a2 2 0 00-2 2v12a2 2 0 002 2z" />
50          </svg>
51        </div>
52      )}
53    </div>
54  )
55}
