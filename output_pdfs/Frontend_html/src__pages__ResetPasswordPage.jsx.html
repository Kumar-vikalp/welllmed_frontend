1import { useState } from 'react';
2import { Link, useNavigate } from 'react-router-dom';
3import { motion } from 'framer-motion';
4import { useUser } from '../contexts/UserContext';
5import Toast from '../components/Toast';
6import { Pill, Eye, EyeOff, Star, Zap, Shield } from 'lucide-react';
7
8export default function ResetPasswordPage() {
9  const [formData, setFormData] = useState({ email: '', otp: '', new_password: '' });
10  const [showPassword, setShowPassword] = useState(false);
11  const [loading, setLoading] = useState(false);
12  const [toast, setToast] = useState({ message: '', type: 'info' });
13  const { resetPassword } = useUser();
14  const navigate = useNavigate();
15
16  const handleChange = (e) => {
17    setFormData({ ...formData, [e.target.name]: e.target.value });
18  };
19
20  const getPasswordStrength = (password) => {
21    if (password.length === 0) return { strength: 0, label: '' };
22    if (password.length < 6) return { strength: 25, label: 'WEAK' };
23    if (password.length < 8) return { strength: 50, label: 'FAIR' };
24    if (password.length >= 8 && /[A-Z]/.test(password) && /[0-9]/.test(password)) {
25      return { strength: 100, label: 'STRONG' };
26    }
27    return { strength: 75, label: 'GOOD' };
28  };
29
30  const passwordStrength = getPasswordStrength(formData.new_password);
31
32  const handleSubmit = async (e) => {
33    e.preventDefault();
34    if (formData.new_password.length < 6) {
35      setToast({ message: 'PASSWORD MUST BE AT LEAST 6 CHARACTERS LONG.', type: 'error' });
36      return;
37    }
38    
39    setLoading(true);
40    setToast({ message: '', type: 'info' });
41
42    try {
43      const response = await resetPassword(formData.email, formData.otp, formData.new_password);
44      setToast({ message: (response.message || 'PASSWORD HAS BEEN RESET SUCCESSFULLY! REDIRECTING TO LOGIN...').toUpperCase(), type: 'success' });
45      setTimeout(() => navigate('/login'), 2000);
46    } catch (err) {
47      const errorMessage = (err.response?.data?.detail || 'FAILED TO RESET PASSWORD. PLEASE CHECK YOUR DETAILS AND TRY AGAIN.').toUpperCase();
48      setToast({ message: errorMessage, type: 'error' });
49    } finally {
50      setLoading(false);
51    }
52  };
53
54  return (
55    <div className="min-h-screen bg-[#FFFDF5] relative overflow-hidden">
56      <Toast message={toast.message} type={toast.type} duration={5000} />
57      
58      {/* Background Pattern */}
59      <div 
60        className="absolute inset-0 opacity-10"
61        style={{
62          backgroundImage: 'radial-gradient(#000 1.5px, transparent 1.5px)',
63          backgroundSize: '20px 20px'
64        }}
65      ></div>
66      
67      {/* Floating Decorative Elements */}
68      <div className="absolute top-10 left-4 sm:top-20 sm:left-10 rotate-12">
69        <Star className="w-6 h-6 sm:w-8 sm:h-8 fill-[#FF6B6B] text-[#FF6B6B] animate-spin-slow" />
70      </div>
71      <div className="absolute top-32 right-4 sm:top-40 sm:right-20 -rotate-12">
72        <Zap className="w-8 h-8 sm:w-12 sm:h-12 fill-[#FFD93D] text-[#FFD93D] animate-bounce" />
73      </div>
74      <div className="absolute bottom-20 left-8 sm:bottom-32 sm:left-32 rotate-45">
75        <Shield className="w-6 h-6 sm:w-10 sm:h-10 fill-[#C4B5FD] text-[#C4B5FD]" />
76      </div>
77
78      <div className="relative z-10 min-h-screen flex items-center justify-center p-4">
79        <motion.div 
80          initial={{ opacity: 0, y: -20 }}
81          animate={{ opacity: 1, y: 0 }}
82          className="w-full max-w-md"
83        >
84          {/* Header */}
85          <div className="text-center mb-8">
86            <Link to="/" className="inline-block mb-6">
87              <div className="bg-[#FF6B6B] border-4 border-black p-4 shadow-[6px_6px_0px_0px_#000] rotate-1 flex items-center gap-3">
88                <Pill className="w-8 h-8 stroke-[3px] text-white" />
89                <span className="text-2xl sm:text-3xl font-black uppercase tracking-tight text-white">
90                  WELLMED
91                </span>
92              </div>
93            </Link>
94            <h2 className="text-2xl sm:text-3xl font-black uppercase tracking-tight mb-4 -rotate-1">
95              RESET PASSWORD
96            </h2>
97            <div className="bg-[#FFD93D] border-4 border-black shadow-[4px_4px_0px_0px_#000] p-4 rotate-1 inline-block">
98              <p className="font-bold uppercase tracking-wide text-sm">
99                ENTER YOUR EMAIL, OTP, AND NEW PASSWORD
100              </p>
101            </div>
102          </div>
103
104          {/* Reset Form */}
105          <div className="bg-white border-4 border-black shadow-[12px_12px_0px_0px_#000] p-6 sm:p-8 -rotate-1 overflow-hidden">
106            <form onSubmit={handleSubmit} className="space-y-6">
107              {/* Email Field */}
108              <div>
109                <label htmlFor="email" className="block font-black uppercase tracking-widest text-xs mb-2">
110                  EMAIL ADDRESS *
111                </label>
112                <input
113                  id="email"
114                  name="email"
115                  type="email"
116                  required
117                  value={formData.email}
118                  onChange={handleChange}
119                  className="w-full border-4 border-black bg-white font-bold text-base h-14 px-4
120                    placeholder:text-black/40 placeholder:font-bold placeholder:uppercase
121                    focus-visible:bg-[#FFD93D] focus-visible:shadow-[4px_4px_0px_0px_#000]
122                    focus-visible:outline-none focus-visible:ring-0
123                    transition-all duration-100"
124                  placeholder="YOUR EMAIL ADDRESS"
125                />
126              </div>
127
128              {/* OTP Field */}
129              <div>
130                <label htmlFor="otp" className="block font-black uppercase tracking-widest text-xs mb-2">
131                  OTP CODE *
132                </label>
133                <input
134                  id="otp"
135                  name="otp"
136                  type="text"
137                  required
138                  value={formData.otp}
139                  onChange={handleChange}
140                  className="w-full border-4 border-black bg-white font-bold text-base h-14 px-4
141                    placeholder:text-black/40 placeholder:font-bold placeholder:uppercase
142                    focus-visible:bg-[#FFD93D] focus-visible:shadow-[4px_4px_0px_0px_#000]
143                    focus-visible:outline-none focus-visible:ring-0
144                    transition-all duration-100"
145                  placeholder="OTP FROM EMAIL"
146                />
147              </div>
148
149              {/* New Password Field */}
150              <div>
151                <label htmlFor="new_password" className="block font-black uppercase tracking-widest text-xs mb-2">
152                  NEW PASSWORD *
153                </label>
154                <div className="relative">
155                  <input
156                    id="new_password"
157                    name="new_password"
158                    type={showPassword ? "text" : "password"}
159                    required
160                    value={formData.new_password}
161                    onChange={handleChange}
162                    className="w-full border-4 border-black bg-white font-bold text-base h-14 px-4 pr-16
163                      placeholder:text-black/40 placeholder:font-bold placeholder:uppercase
164                      focus-visible:bg-[#FFD93D] focus-visible:shadow-[4px_4px_0px_0px_#000]
165                      focus-visible:outline-none focus-visible:ring-0
166                      transition-all duration-100"
167                    placeholder="NEW PASSWORD"
168                  />
169                  <button
170                    type="button"
171                    onClick={() => setShowPassword(!showPassword)}
172                    className="absolute right-4 top-1/2 transform -translate-y-1/2 bg-white border-2 border-black p-2
173                      hover:bg-[#C4B5FD] transition-colors duration-100"
174                  >
175                    {showPassword ? (
176                      <EyeOff className="w-4 h-4 stroke-[3px] text-black" />
177                    ) : (
178                      <Eye className="w-4 h-4 stroke-[3px] text-black" />
179                    )}
180                  </button>
181                </div>
182
183                {/* Password Strength Indicator */}
184                {formData.new_password && (
185                  <div className="mt-3">
186                    <div className="flex items-center justify-between mb-2">
187                      <span className="font-black uppercase tracking-widest text-xs">PASSWORD STRENGTH</span>
188                      <span className={`font-black uppercase tracking-widest text-xs px-2 py-1 border-2 border-black ${
189                        passwordStrength.strength === 100 ? 'bg-[#FFD93D]' :
190                        passwordStrength.strength >= 75 ? 'bg-[#C4B5FD]' :
191                        passwordStrength.strength >= 50 ? 'bg-[#FFD93D]' : 'bg-[#FF6B6B] text-white'
192                      }`}>
193                        {passwordStrength.label}
194                      </span>
195                    </div>
196                    <div className="w-full bg-white border-4 border-black h-3 overflow-hidden">
197                      <div 
198                        className={`h-full transition-all duration-300 ${
199                          passwordStrength.strength === 100 ? 'bg-[#FFD93D]' :
200                          passwordStrength.strength >= 75 ? 'bg-[#C4B5FD]' :
201                          passwordStrength.strength >= 50 ? 'bg-[#FFD93D]' : 'bg-[#FF6B6B]'
202                        }`}
203                        style={{ width: `${passwordStrength.strength}%` }}
204                      ></div>
205                    </div>
206                  </div>
207                )}
208              </div>
209              
210              {/* Submit Button */}
211              <button
212                type="submit"
213                disabled={loading}
214                className="w-full bg-[#FF6B6B] border-4 border-black font-black uppercase tracking-widest 
215                  text-sm py-4 h-14 disabled:opacity-50 disabled:cursor-not-allowed
216                  shadow-[6px_6px_0px_0px_#000]
217                  hover:shadow-[3px_3px_0px_0px_#000] hover:translate-x-[3px] hover:translate-y-[3px]
218                  active:shadow-none active:translate-x-[6px] active:translate-y-[6px]
219                  transition-all duration-100"
220              >
221                {loading ? (
222                  <div className="flex items-center justify-center">
223                    <div className="w-4 h-4 border-2 border-white mr-2 animate-spin"></div>
224                    RESETTING PASSWORD...
225                  </div>
226                ) : (
227                  'RESET PASSWORD'
228                )}
229              </button>
230            </form>
231          </div>
232
233          {/* Back to Login Link */}
234          <div className="text-center mt-8">
235            <div className="bg-[#C4B5FD] border-4 border-black p-4 shadow-[6px_6px_0px_0px_#000] rotate-1">
236              <p className="font-bold uppercase tracking-wide text-sm">
237                REMEMBERED YOUR PASSWORD?{' '}
238                <Link 
239                  to="/login" 
240                  className="font-black underline decoration-4 underline-offset-4 hover:text-[#FF6B6B] transition-colors"
241                >
242                  BACK TO LOGIN
243                </Link>
244              </p>
245            </div>
246          </div>
247
248          {/* Security Tips */}
249          <div className="mt-8">
250            <div className="bg-black border-4 border-black shadow-[8px_8px_0px_0px_#000] p-6 -rotate-1">
251              <h3 className="font-black uppercase tracking-widest text-sm text-white mb-4 border-b-2 border-white pb-2">
252                PASSWORD SECURITY TIPS
253              </h3>
254              <div className="space-y-2 text-white">
255                <div className="flex items-center gap-3">
256                  <div className="w-2 h-2 bg-[#FFD93D]"></div>
257                  <span className="font-bold text-sm uppercase">USE AT LEAST 8 CHARACTERS</span>
258                </div>
259                <div className="flex items-center gap-3">
260                  <div className="w-2 h-2 bg-[#FFD93D]"></div>
261                  <span className="font-bold text-sm uppercase">INCLUDE UPPERCASE & LOWERCASE</span>
262                </div>
263                <div className="flex items-center gap-3">
264                  <div className="w-2 h-2 bg-[#FFD93D]"></div>
265                  <span className="font-bold text-sm uppercase">ADD NUMBERS & SYMBOLS</span>
266                </div>
267                <div className="flex items-center gap-3">
268                  <div className="w-2 h-2 bg-[#FFD93D]"></div>
269                  <span className="font-bold text-sm uppercase">AVOID COMMON WORDS</span>
270                </div>
271              </div>
272            </div>
273          </div>
274        </motion.div>
275      </div>
276    </div>
277  );
278}
