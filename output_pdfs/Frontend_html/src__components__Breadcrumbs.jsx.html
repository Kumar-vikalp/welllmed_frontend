1import { Link, useLocation } from 'react-router-dom';
2import { useSelector } from 'react-redux';
3import { selectCurrentProduct } from '../store/slices/productsSlice';
4
5export default function Breadcrumbs() {
6  const location = useLocation();
7  const currentProduct = useSelector(selectCurrentProduct);
8  const pathnames = location.pathname.split('/').filter((x) => x);
9
10  const breadcrumbNameMap = {
11    'products': 'Products',
12    'product': 'Product Details',
13    'cart': 'Shopping Cart',
14    'checkout': 'Checkout',
15    'profile': 'My Profile',
16    'orders': 'Order History',
17    'search': 'Search',
18    'generic-info': 'Generic Medicines',
19    'about': 'About Us',
20    'contact': 'Contact Us',
21    'privacy': 'Privacy Policy',
22    'terms': 'Terms of Service',
23    'help': 'Help Center',
24    'how-to-order': 'How to Order'
25  };
26
27  // Get display label for each breadcrumb
28  const getDisplayName = (name, index) => {
29    const isProductSlug =
30      pathnames[index - 1] === 'product' &&
31      currentProduct &&
32      index === pathnames.length - 1;
33
34    if (isProductSlug) {
35      return currentProduct.name;
36    }
37
38    if (breadcrumbNameMap[name]) {
39      return breadcrumbNameMap[name];
40    }
41
42    // Fallback: Humanize the slug
43    return name
44      .split('-')
45      .map((part) => part.charAt(0).toUpperCase() + part.slice(1))
46      .join(' ');
47  };
48
49  // Build breadcrumb path route
50  const getRouteForBreadcrumb = (name, index) => {
51    // Special handling for product routes
52    if (pathnames[index - 1] === 'product' && index === pathnames.length - 1) {
53      return `/${pathnames.slice(0, index + 1).join('/')}`;
54    }
55    return `/${pathnames.slice(0, index + 1).join('/')}`;
56  };
57
58  // Don't render on homepage
59  if (pathnames.length === 0) return null;
60
61  return (
62    <nav className="bg-gray-50 border-b border-gray-200">
63      <div className="max-w-7xl mx-auto px-4 py-3">
64        <ol className="flex items-center space-x-2 text-sm">
65          <li>
66            <Link
67              to="/"
68              className="text-gray-500 hover:text-purple-600 transition-colors"
69            >
70              <svg className="w-4 h-4" fill="currentColor" viewBox="0 0 20 20">
71                <path d="M10.707 2.293a1 1 0 00-1.414 0l-7 7a1 1 0 001.414 1.414L4 10.414V17a1 1 0 001 1h2a1 1 0 001-1v-2a1 1 0 011-1h2a1 1 0 011 1v2a1 1 0 001 1h2a1 1 0 001-1v-6.586l.293.293a1 1 0 001.414-1.414l-7-7z" />
72              </svg>
73            </Link>
74          </li>
75          {pathnames.map((name, index) => {
76            const routeTo = getRouteForBreadcrumb(name, index);
77            const isLast = index === pathnames.length - 1;
78            const displayName = getDisplayName(name, index);
79
80            return (
81              <li key={name} className="flex items-center">
82                <svg
83                  className="w-4 h-4 text-gray-400 mx-2"
84                  fill="currentColor"
85                  viewBox="0 0 20 20"
86                >
87                  <path
88                    fillRule="evenodd"
89                    d="M7.293 14.707a1 1 0 010-1.414L10.586 10 7.293 6.707a1 1 0 011.414-1.414l4 4a1 1 0 010 1.414l-4 4a1 1 0 01-1.414 0z"
90                    clipRule="evenodd"
91                  />
92                </svg>
93                {isLast ? (
94                  <span className="text-gray-900 font-medium">
95                    {displayName}
96                  </span>
97                ) : (
98                  <Link
99                    to={routeTo}
100                    className="text-purple-600 hover:text-purple-800 transition-colors"
101                  >
102                    {displayName}
103                  </Link>
104                )}
105              </li>
106            );
107          })}
108        </ol>
109      </div>
110    </nav>
111  );
112}
