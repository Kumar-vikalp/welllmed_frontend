src/components/ProductCard.jsxJSX
1import { motion } from 'framer-motion';
2import { Link } from 'react-router-dom';
3import { useDispatch, useSelector } from 'react-redux';
4import { addToCart, fetchCart, selectCartSyncing } from '../store/slices/cartSlice';
5import { selectUser } from '../store/slices/userSlice';
6import { useState, memo } from 'react';
7import LazyImage from './LazyImage';
8import Card from './Card';
9import Button from './Button';
10
11const ProductCard = memo(function ProductCard({ product }) {
12  const dispatch = useDispatch();
13  const user = useSelector(selectUser);
14  const syncing = useSelector(selectCartSyncing);
15  const [isAdding, setIsAdding] = useState(false);
16  const discountedPrice = product.discounted_price;
17
18  const handleAddToCart = async (e) => {
19    e.preventDefault();
20    e.stopPropagation();
21    
22    if (!user) {
23      alert('Please login to add items to cart');
24      return;
25    }
26    
27    if (product.available_stock <= 0) return;
28    
29    setIsAdding(true);
30    try {
31      // Add to cart via API
32      await dispatch(addToCart({ 
33        product_id: product.product_id, 
34        quantity: 1 
35      })).unwrap();
36      
37      // Refresh cart to get updated data
38      await dispatch(fetchCart());
39      
40    } catch (error) {
41      console.error('Failed to add to cart:', error);
42      alert('Failed to add item to cart. Please try again.');
43    } finally {
44      setIsAdding(false);
45    }
46  };
47
48  return (
49    <Card
50      layout
51      className="overflow-hidden flex flex-col"
52      rotation={Math.random() > 0.7 ? (Math.random() > 0.5 ? 1 : -1) : 0}
53    >
54      <Link to={`/product/${product.slug}`} className="block flex-grow" onClick={(e) => e.stopPropagation()}>
55        <div className="relative">
56          <LazyImage 
57            src={product.images[0]} 
58            alt={product.name} 
59            className="w-full h-32 md:h-48 object-cover border-b-4 border-neo-ink" 
60          />
61          {product.trending && (
62            <span className="absolute -top-2 -left-2 neo-badge bg-neo-accent rotate-12 z-10">
63              🔥 HOT
64            </span>
65          )}
66          {product.discount > 0 && (
67            <span className="absolute -top-2 -right-2 neo-badge bg-neo-secondary -rotate-12 z-10">
68              {product.discount}% OFF
69            </span>
70          )}
71        </div>
72        <div className="p-3 md:p-4 flex-grow">
73          <p className="text-xs md:text-sm font-bold uppercase tracking-wide">{product.company}</p>
74          <h3 className="text-sm md:text-base font-black mt-1 mb-2 line-clamp-2 leading-tight">
75            {product.name}
76          </h3>
77          <div className="flex justify-between items-center mb-3">
78            <div>
79              <p className="text-lg md:text-xl font-black">₹{discountedPrice.toFixed(2)}</p>
80              {product.discount > 0 && (
81                <p className="text-xs md:text-sm font-bold line-through opacity-60">₹{product.mrp.toFixed(2)}</p>
82              )}
83            </div>
84            <span className={`neo-badge rotate-2 ${
85              product.available_stock > 0 
86                ? 'bg-neo-secondary' 
87                : 'bg-neo-accent'
88            }`}>
89              {product.available_stock > 0 ? 'IN STOCK' : 'OUT OF STOCK'}
90            </span>
91          </div>
92        </div>
93      </Link>
94      
95      {/* Add to Cart Button */}
96      <div className="p-3 md:p-4 pt-0">
97        <Button
98          variant={product.available_stock > 0 && user ? 'primary' : 'outline'}
99          size="sm"
100          className="w-full"
101          onClick={handleAddToCart}
102          disabled={product.available_stock <= 0 || isAdding || syncing || !user}
103        >
104          {isAdding || syncing ? (
105            'ADDING...'
106          ) : !user ? (
107            'LOGIN TO ADD'
108          ) : product.available_stock > 0 ? (
109            'ADD TO CART'
110          ) : (
111            'OUT OF STOCK'
112          )}
113        </Button>
114      </div>
115    </Card>
116  );
117});
118
119export default ProductCard;
