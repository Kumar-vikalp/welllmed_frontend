1import { Outlet } from 'react-router-dom';
2import { Suspense } from 'react';
3import { motion } from 'framer-motion';
4import Navbar from './Navbar';
5import Footer from './Footer';
6import Breadcrumbs from './Breadcrumbs';
7import { useLocation } from 'react-router-dom';
8
9// Loading component for pages
10const PageLoader = () => (
11  <motion.div 
12    initial={{ opacity: 0 }}
13    animate={{ opacity: 1 }}
14    className="flex items-center justify-center py-20"
15  >
16    <div className="animate-spin rounded-full h-12 w-12 border-b-2 border-purple-600"></div>
17  </motion.div>
18);
19
20export default function Layout() {
21  const location = useLocation();
22
23  // Don't show breadcrumbs on home page or auth pages
24  const isAuthPage = location.pathname === '/login' ||
25                     location.pathname === '/signup' ||
26                     location.pathname === '/forgot-password' ||
27                     location.pathname === '/reset-password';
28  const showBreadcrumbs = location.pathname !== '/' && !isAuthPage;
29
30  return (
31    <div className="flex flex-col min-h-screen bg-gray-50 pt-0 md:pt-16">
32      <Navbar />
33      {showBreadcrumbs && <Breadcrumbs />}
34      <main className="flex-grow pb-20 md:pb-8 min-h-screen">
35        <Suspense fallback={<PageLoader />}>
36          <Outlet />
37        </Suspense>
38      </main>
39      <Footer />
40    </div>
41  );
42}
