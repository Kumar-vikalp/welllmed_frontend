1import { createContext, useContext, useState, useEffect, useCallback } from 'react';
2import api from '../api/axiosConfig';
3import { useUser } from './UserContext';
4
5const OrderContext = createContext();
6
7export function OrderProvider({ children }) {
8  const [orders, setOrders] = useState([]);
9  const [loading, setLoading] = useState(false);
10  const { user } = useUser();
11
12  // Memoize fetchOrders to prevent unnecessary re-creations
13  const fetchOrders = useCallback(async () => {
14    if (!user) return;
15
16    try {
17      setLoading(true);
18      const response = await api.get('/orders/');
19      
20      // Handle different response formats
21      let ordersData = [];
22      if (response.data) {
23        if (Array.isArray(response.data)) {
24          ordersData = response.data;
25        } else if (response.data.results && Array.isArray(response.data.results)) {
26          ordersData = response.data.results;
27        } else if (typeof response.data === 'object') {
28          ordersData = [response.data];
29        }
30      }
31
32      const transformedOrders = ordersData.map(order => ({
33        orderId: order.order_id,
34        date: order.created_at,
35        total: parseFloat(order.total),
36        status: order.status,
37        paymentMethod: order.payment_method,
38        estimatedDelivery: order.estimated_delivery,
39        items: order.items || [],
40        canCancel: order.can_cancel,
41        deliveryDetails: {
42          name: user.name,
43          address: user.address,
44          phone: user.phone,
45          estimatedDelivery: order.estimated_delivery
46        }
47      }));
48
49      setOrders(transformedOrders);
50    } catch (error) {
51      console.error('Failed to fetch orders', error);
52      // Set empty array on error instead of throwing
53      setOrders([]);
54    } finally {
55      setLoading(false);
56    }
57  }, [user]);
58
59  // Effect now depends on user and memoized fetchOrders
60  useEffect(() => {
61    if (user) {
62      fetchOrders();
63    } else {
64      setOrders([]);
65    }
66  }, [user, fetchOrders]);
67
68  const createOrder = async (orderData) => {
69    if (!user) {
70      throw new Error('Please login to place an order');
71    }
72
73    try {
74      const orderPayload = {
75        payment_method: orderData.paymentMethod || 'UPI',
76        items: orderData.items || []
77      };
78
79      const response = await api.post('/orders/create/', orderPayload, {
80        headers: { 'Content-Type': 'application/json' },
81      });
82
83      // Refresh orders after creating
84      await fetchOrders();
85      return response.data;
86    } catch (error) {
87      console.error('Failed to create order', error);
88      throw error;
89    }
90  };
91
92  const cancelOrder = async (orderId) => {
93    if (!user) return;
94
95    try {
96      await api.post(`/orders/${orderId}/cancel/`);
97      await fetchOrders();
98    } catch (error) {
99      console.error('Failed to cancel order', error);
100      throw error;
101    }
102  };
103
104  const getOrderById = async (orderId) => {
105    if (!user) return null;
106
107    try {
108      const response = await api.get(`/orders/${orderId}/`);
109      return response.data;
110    } catch (error) {
111      console.error('Failed to fetch order details', error);
112      throw error;
113    }
114  };
115
116  // Keep addOrder for backward compatibility
117  const addOrder = (order) => {
118    setOrders(prevOrders => [order, ...prevOrders]);
119  };
120
121  const value = {
122    orders,
123    loading,
124    addOrder,
125    createOrder,
126    cancelOrder,
127    getOrderById,
128    fetchOrders
129  };
130
131  return (
132    <OrderContext.Provider value={value}>
133      {children}
134    </OrderContext.Provider>
135  );
136}
137
138export const useOrders = () => useContext(OrderContext);
