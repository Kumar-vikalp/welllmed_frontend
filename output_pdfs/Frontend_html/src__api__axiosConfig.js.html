1import axios from 'axios';
2
3const api = axios.create({
4  baseURL: 'https://vello-alpha.vercel.app/api',
5  timeout: 15000, // 15 second timeout
6  headers: {
7    'Content-Type': 'application/json',
8    'Accept': 'application/json',
9  },
10});
11
12// Interceptor to add the access token to every request
13api.interceptors.request.use(
14  (config) => {
15    const token = localStorage.getItem('access_token');
16    if (token) {
17      config.headers['Authorization'] = `Bearer ${token}`;
18    }
19    return config;
20  },
21  (error) => {
22    return Promise.reject(error);
23  }
24);
25
26// Interceptor to handle token refreshment on 401 Unauthorized errors
27api.interceptors.response.use(
28  (response) => {
29    return response;
30  },
31  async (error) => {
32    // Handle timeout errors
33    if (error.code === 'ECONNABORTED') {
34      console.error('Request timeout');
35      return Promise.reject(new Error('Request timeout. Please try again.'));
36    }
37    
38    if (!error.response) {
39      console.error('Network error');
40      return Promise.reject(new Error('Network error. Please check your connection.'));
41    }
42    
43    const originalRequest = error.config;
44    // Check if the error is 401 and it's not a retry request
45    if (error.response.status === 401 && !originalRequest._retry) {
46      originalRequest._retry = true;
47      const refreshToken = localStorage.getItem('refresh_token');
48
49      if (refreshToken) {
50        try {
51          const response = await axios.post('https://vello-alpha.vercel.app/api/token/refresh/', {
52            refresh: refreshToken,
53          });
54          const newAccessToken = response.data.access;
55
56          localStorage.setItem('access_token', newAccessToken);
57          
58          // Update the authorization header for the original request
59          api.defaults.headers.common['Authorization'] = `Bearer ${newAccessToken}`;
60          originalRequest.headers['Authorization'] = `Bearer ${newAccessToken}`;
61          
62          return api(originalRequest);
63        } catch (refreshError) {
64          // If refresh fails, log out the user
65          console.error("Token refresh failed", refreshError);
66          // Here you might want to trigger a logout action
67          localStorage.removeItem('access_token');
68          localStorage.removeItem('refresh_token');
69          localStorage.removeItem('user');
70          window.location.href = '/login';
71          return Promise.reject(refreshError);
72        }
73      }
74    }
75    return Promise.reject(error);
76  }
77);
78
79export default api;
