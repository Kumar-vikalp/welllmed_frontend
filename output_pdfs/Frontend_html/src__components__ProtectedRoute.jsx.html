1import { Navigate, useLocation } from 'react-router-dom';
2import { useSelector } from 'react-redux';
3import { selectUser } from '../store/slices/userSlice';
4
5export default function ProtectedRoute({ children }) {
6  const user = useSelector(selectUser);
7  const location = useLocation();
8
9  if (!user) {
10    // Redirect them to the /login page, but save the current location they were
11    // trying to go to. This allows us to send them along to that page after they login.
12    return <Navigate to="/login" state={{ from: location }} replace />;
13  }
14
15  return children;
16}
