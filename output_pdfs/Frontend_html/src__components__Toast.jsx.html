1import { useState, useEffect } from 'react';
2import { motion, AnimatePresence } from 'framer-motion';
3
4export default function Toast({ message, type = 'info', duration = 4000 }) {
5  const [isVisible, setIsVisible] = useState(false);
6
7  useEffect(() => {
8    if (message) {
9      setIsVisible(true);
10      const timer = setTimeout(() => {
11        setIsVisible(false);
12      }, duration);
13      return () => clearTimeout(timer);
14    }
15  }, [message, duration]);
16
17  const typeStyles = {
18    success: 'bg-[#C4B5FD] border-black',
19    error: 'bg-[#FF6B6B] border-black text-white',
20    info: 'bg-[#FFD93D] border-black',
21    warning: 'bg-[#FFD93D] border-black'
22  };
23
24  return (
25    <AnimatePresence>
26      {isVisible && message && (
27        <motion.div
28          initial={{ opacity: 0, y: -50, x: '-50%' }}
29          animate={{ opacity: 1, y: 0, x: '-50%' }}
30          exit={{ opacity: 0, y: -50, x: '-50%' }}
31          transition={{ duration: 0.2, ease: 'easeOut' }}
32          className={`fixed top-4 left-1/2 z-50 px-4 sm:px-6 py-3 sm:py-4 border-4 shadow-[4px_4px_0px_0px_#000] sm:shadow-[6px_6px_0px_0px_#000] max-w-sm sm:max-w-md ${typeStyles[type]}`}
33        >
34          <p className="font-black text-xs sm:text-sm uppercase tracking-wide text-center">
35            {message}
36          </p>
37        </motion.div>
38      )}
39    </AnimatePresence>
40  );
41}
